diplomatic_envoys = {
	is_shown = {
		house_head ?= {
			government_allows = administrative
			government_has_flag = government_has_powerful_families
			
			NOR = {
				government_has_flag = government_uses_merit_family_aspirations
				government_is_japanese_trigger = yes
				government_has_flag = government_has_house_blocs
			}
		}
	}

	level = {
		cost = {
			influence = @admin_level_1_cost
		}
		
		powerful_family_top_liege_modifier = {
			negate_diplomacy_penalty_add = 1
			court_grandeur_baseline_add = 1
		}

		powerful_family_member_modifier = {
			domicile_monthly_prestige_mult = 0.1
		}
		
		parameters = {
			diplomatic_envoys = yes
			increased_request_appointment_acceptance_1 = yes
			increased_marriage_acceptance_1 = yes
		}
		
		ai_score = {
			value = 1
			root ?= {
				dynasty ?= {
					if = {
						limit = { has_dynasty_perk = glory_legacy_1 }
						add = 10
					}
					if = {
						limit = { has_dynasty_perk = ep1_culture_legacy_1 }
						add = 10
					}
				}
				if = {
					limit = { ai_compassion > high_positive_ai_value }
					add = {
						value = ai_compassion
						multiply = 0.05
					}
				}
				if = {
					limit = { ai_sociability > 0 }
					add = {
						value = ai_sociability 
						multiply = 0.05
					}
				}
				if = {
					limit = { ai_honor < 0 }
					add = {
						value = ai_honor 
						multiply = 0.05
					}
				}
			}
		}
	}

	level = {
		cost = {
			influence = @admin_level_2_cost
		}
		
		powerful_family_top_liege_modifier = {
			negate_diplomacy_penalty_add = 1
			court_grandeur_baseline_add = 2
		}

		powerful_family_member_modifier = {
			domicile_monthly_prestige_mult = 0.2
		}
		
		parameters = {
			diplomatic_envoys = yes
			increased_request_appointment_acceptance_2 = yes
			increased_marriage_acceptance_2 = yes
		}
		
		ai_score = {
			value = 400
		}
	}

	level = {
		cost = {
			influence = @admin_level_3_cost
		}
		
		powerful_family_top_liege_modifier = {
			negate_diplomacy_penalty_add = 1
			court_grandeur_baseline_add = 4
		}

		powerful_family_member_modifier = {
			domicile_monthly_prestige_mult = 0.3
			ignore_negative_culture_opinion = yes
