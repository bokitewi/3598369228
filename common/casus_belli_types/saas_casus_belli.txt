# Spring and Autumn situation wars.

saas_tianzi_force_tributary_cb = {
	icon = make_tributary_cb
	group = subjugation
	ai_only_against_neighbors = no

	allowed_for_character = {
		has_title = title:e_huaxia
		saas_is_ritual_restoration_phase_trigger = yes
	}
	allowed_against_character = {
		scope:defender = {
			is_independent_ruler = yes
			is_landed = yes
			saas_is_lijiao_ruler_trigger = yes
			NOT = { is_tributary_of = scope:attacker }
			NOT = { government_has_flag = cannot_be_vassal_or_liege }
		}
	}
	should_invalidate = {
		OR = {
			NOT = { scope:attacker = { has_title = title:e_huaxia } }
			NOT = { scope:attacker = { saas_is_ritual_restoration_phase_trigger = yes } }
			NOT = { scope:defender = { is_independent_ruler = yes } }
			scope:defender = { is_tributary_of = scope:attacker }
		}
	}

	cost = { prestige = { value = 0 } }
	on_declaration = { on_declared_war = yes }
	on_invalidated_desc = msg_make_tributary_war_invalidated_message
	on_invalidated = {}

	on_victory_desc = { desc = saas_tianzi_force_tributary_victory_desc }
	on_victory = {
		start_tributary_from_war_effect = {
			TRIBUTARY = scope:defender
			SUZERAIN = scope:attacker
			WAR = root.war
		}
		scope:attacker = {
			add_prestige = major_prestige_value
			add_legitimacy = 20
		}
		scope:defender = { add_prestige = { value = major_prestige_value multiply = -1 } }
		add_truce_attacker_victory_effect = yes
		saas_trigger_catalyst_effect = {
			CATALYST = saas_catalyst_tribute_established
			CHARACTER = scope:attacker
		}
	}

	on_white_peace_desc = { desc = saas_tianzi_force_tributary_white_peace_desc }
	on_white_peace = {
		scope:attacker = { add_prestige = { value = medium_prestige_value multiply = -1 } }
		add_truce_white_peace_effect = yes
	}
	on_defeat_desc = { desc = saas_tianzi_force_tributary_defeat_desc }
	on_defeat = {
		scope:attacker = {
			add_prestige = { value = major_prestige_value multiply = -1 }
			add_legitimacy = -20
			save_temporary_scope_as = loser
		}
		add_truce_attacker_defeat_effect = yes
		on_lost_aggression_war_discontent_loss = yes
		saas_trigger_catalyst_effect = {
			CATALYST = saas_catalyst_tianzi_defeat
			CHARACTER = scope:attacker
		}
	}

	on_primary_attacker_death = invalidate
	on_primary_defender_death = inherit
	attacker_allies_inherit = yes
	defender_allies_inherit = yes
	transfer_behavior = transfer
	war_name = "SAAS_TIANZI_FORCE_TRIBUTARY_WAR_NAME"
	cb_name = "SAAS_TIANZI_FORCE_TRIBUTARY_CB_NAME"
	interface_priority = 150
	ticking_war_score_targets_entire_realm = yes
	attacker_wargoal_percentage = 0.8
	max_defender_score_from_occupation = 150
	max_attacker_score_from_occupation = 150
	ai_score_mult = { value = 1 }
}

saas_wangdao_kingdom_conquest_cb = {
	icon = religious
	group = conquest
	combine_into_one = yes
	should_show_war_goal_subview = yes
	mutually_exclusive_titles = { always = yes }
	target_title_tier = kingdom
	target_titles = all
	target_de_jure_regions_above = yes
	ignore_effect = change_title_holder
	ai_can_target_all_titles = { always = yes }

	allowed_for_character = {
		saas_is_ritual_restoration_phase_trigger = yes
		saas_is_wangdao_trigger = yes
		is_landed = yes
	}
	allowed_against_character = {
		scope:defender = {
			is_independent_ruler = yes
			is_landed = yes
			NOT = { saas_is_lijiao_ruler_trigger = yes }
		}
	}
	valid_to_start = {
		scope:target.tier = tier_kingdom
		scope:target = {
			any_in_de_jure_hierarchy = {
				tier = tier_county
				holder ?= { target_is_liege_or_above = scope:defender }
			}
		}
	}
	should_invalidate = {
		OR = {
			NOT = { scope:attacker = { saas_is_ritual_restoration_phase_trigger = yes } }
			NOT = { scope:attacker = { saas_is_wangdao_trigger = yes } }
			NOT = { scope:defender = { is_independent_ruler = yes } }
		}
	}

	cost = {
		prestige = {
			value = 300
			multiply = common_cb_prestige_cost_multiplier
		}
	}
	on_declaration = { on_declared_war = yes }
	on_invalidated_desc = msg_religious_war_invalidation_region_message
	on_invalidated = {}

	on_victory_desc = { desc = saas_wangdao_kingdom_conquest_victory_desc }
	on_victory = {
		create_title_and_vassal_change = {
			type = conquest
			save_scope_as = change
			add_claim_on_loss = yes
		}
		hidden_effect = {
			setup_invasion_cb = {
				titles = target_titles
				attacker = scope:attacker
				defender = scope:defender
				change = scope:change
				take_occupied = no
			}
		}
		resolve_title_and_vassal_change = scope:change

		if = {
			limit = { scope:defender.prestige > 0 }
			scope:attacker = { add_prestige = { value = scope:defender.prestige multiply = 0.8 } }
			scope:defender = { add_prestige = { value = prestige multiply = -0.8 } }
		}
		if = {
			limit = { scope:defender.piety > 0 }
			scope:attacker = { add_piety = { value = scope:defender.piety multiply = 0.8 } }
			scope:defender = { add_piety = { value = piety multiply = -0.8 } }
		}
		add_truce_attacker_victory_effect = yes
	}
	on_white_peace_desc = { desc = saas_wangdao_kingdom_conquest_white_peace_desc }
	on_white_peace = {
		scope:attacker = { add_prestige = { value = medium_prestige_value multiply = -1 } }
		add_truce_white_peace_effect = yes
	}
	on_defeat_desc = { desc = saas_wangdao_kingdom_conquest_defeat_desc }
	on_defeat = {
		scope:attacker = {
			add_prestige = { value = major_prestige_value multiply = -1 }
			save_temporary_scope_as = loser
		}
		add_truce_attacker_defeat_effect = yes
		on_lost_aggression_war_discontent_loss = yes
	}

	on_primary_attacker_death = inherit
	on_primary_defender_death = inherit
	attacker_allies_inherit = yes
	defender_allies_inherit = yes
	transfer_behavior = transfer
	war_name = "SAAS_WANGDAO_KINGDOM_WAR_NAME"
	cb_name = "SAAS_WANGDAO_KINGDOM_CB_NAME"
	interface_priority = 140
	use_de_jure_wargoal_only = yes
	attacker_wargoal_percentage = 0.8
	max_defender_score_from_occupation = 150
	max_attacker_score_from_occupation = 150
	ai_score_mult = { value = 1 }
}

saas_warring_states_conquest_cb = {
	icon = invasion
	group = invasion
	combine_into_one = yes
	should_show_war_goal_subview = yes
	mutually_exclusive_titles = { always = yes }
	target_title_tier = kingdom
	target_titles = neighbor_land_or_water
	target_de_jure_regions_above = yes
	ignore_effect = change_title_holder

	allowed_for_character = {
		saas_is_warring_states_phase_trigger = yes
		saas_is_badao_trigger = yes
		is_landed = yes
	}
	allowed_against_character = {
		scope:defender = {
			is_independent_ruler = yes
			is_landed = yes
		}
	}
	valid_to_start = { scope:target.tier = tier_kingdom }
	should_invalidate = {
		OR = {
			NOT = { scope:attacker = { saas_is_warring_states_phase_trigger = yes } }
			NOT = { scope:attacker = { saas_is_badao_trigger = yes } }
			NOT = { scope:defender = { is_independent_ruler = yes } }
		}
	}

	cost = {
		prestige = {
			value = 500
			multiply = common_cb_prestige_cost_multiplier
		}
	}
	on_declaration = { on_declared_war = yes }
	on_invalidated_desc = msg_invasion_war_invalidated_message
	on_invalidated = {}

	on_victory_desc = { desc = saas_warring_states_conquest_victory_desc }
	on_victory = {
		scope:defender = {
			every_realm_de_jure_duchy = {
				limit = {
					any_in_de_jure_hierarchy = {
						percent = 1
						tier = tier_county
						county_controller = scope:attacker
					}
				}
				add_to_temporary_list = saas_fully_occupied_duchies
			}
		}
		if = {
			limit = { any_in_list = { list = saas_fully_occupied_duchies always = yes } }
			create_title_and_vassal_change = {
				type = conquest
				save_scope_as = change
				add_claim_on_loss = yes
			}
			hidden_effect = {
				setup_invasion_cb = {
					titles = saas_fully_occupied_duchies
					attacker = scope:attacker
					defender = scope:defender
					change = scope:change
					take_occupied = yes
				}
			}
			resolve_title_and_vassal_change = scope:change
		}
		scope:attacker = { add_prestige = major_prestige_value }
		add_truce_attacker_victory_effect = yes
	}
	on_white_peace_desc = { desc = saas_warring_states_conquest_white_peace_desc }
	on_white_peace = { add_truce_white_peace_effect = yes }
	on_defeat_desc = { desc = saas_warring_states_conquest_defeat_desc }
	on_defeat = {
		scope:attacker = {
			add_prestige = { value = major_prestige_value multiply = -1 }
			save_temporary_scope_as = loser
		}
		add_truce_attacker_defeat_effect = yes
		on_lost_aggression_war_discontent_loss = yes
	}

	on_primary_attacker_death = inherit
	on_primary_defender_death = inherit
	attacker_allies_inherit = yes
	defender_allies_inherit = yes
	transfer_behavior = transfer
	war_name = "SAAS_WARRING_STATES_CONQUEST_WAR_NAME"
	cb_name = "SAAS_WARRING_STATES_CONQUEST_CB_NAME"
	interface_priority = 145
	attacker_wargoal_percentage = 0.8
	max_defender_score_from_occupation = 150
	max_attacker_score_from_occupation = 150
	ai_score_mult = { value = 1 }
}
