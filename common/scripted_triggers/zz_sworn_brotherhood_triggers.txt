# Oath-breaking detection
is_sworn_brother_oath_breaking_trigger = {
    exists = FROM
    OR = {
        AND = {
            had_relation_blood_brother = { target = FROM }
            OR = {
                is_at_war_with = FROM
                has_started_scheme_against_target = FROM
                is_rival_of = FROM
            }
        }
    }
}
