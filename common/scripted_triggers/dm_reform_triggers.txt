dm_reform_reformer_candidate_trigger = {
	is_alive = yes
	is_adult = yes
	is_imprisoned = no
	NOT = { has_trait = incapable }
	NOT = { this = $RULER$ }
	custom_description = {
		text = dm_reform_reformer_still_eligible_tt
		OR = {
			is_courtier_of = $RULER$
			AND = {
				top_liege = $RULER$.top_liege
				$RULER$ = {
					OR = {
						government_has_flag = government_is_administrative
						government_has_flag = government_is_celestial
					}
				}
			}
		}
	}
}

dm_reform_has_reformer_trigger = {
	exists = var:dm_reform_reformer
	var:dm_reform_reformer = {
		dm_reform_reformer_candidate_trigger = {
			RULER = prev.story_owner
		}
	}
}

dm_reform_japan_defense_mobilization_valid_trigger = {
	# CHARACTER is the interaction actor. This mirrors the loaded TGP trigger
	# while avoiding its law-context ROOT reference in a rootless interaction.
	custom_tooltip = {
		text = defense_mobilization_law_valid_trigger
		OR = {
			AND = {
				is_at_war = yes
				save_temporary_scope_as = liege_temp
				any_character_war = {
					is_civil_war = no
					save_temporary_scope_as = war_temp
					scope:liege_temp = {
						is_leader_in_war = scope:war_temp
						is_defender_in_war = scope:war_temp
					}
					OR = {
						war_attacker_total_strength_value > scope:liege_temp.max_strength_fifty_percent_value
						primary_attacker.max_strength_with_allies_value > scope:liege_temp.max_strength_fifty_percent_value
					}
					primary_attacker = {
						OR = {
							capital_county.empire != scope:liege_temp.capital_county.empire
							NOT = { government_is_japanese_trigger = yes }
						}
					}
				}
			}
			capital_county.empire = {
				any_de_jure_county = {
					holder.top_liege != $CHARACTER$.top_liege
				}
			}
		}
	}
}
