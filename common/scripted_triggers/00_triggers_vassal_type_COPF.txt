# 多数封臣拥有【传统世袭】
most_vassals_have_hereditary_or_above = {
	custom_tooltip = {
		text = tt_most_vassals_have_hereditary_or_above
		OR = {
			any_in_list = {
				variable = list_valid_vassal
				OR = {
					subject_contract_has_flag = succession_rights_hereditary
					subject_contract_has_flag = locked_to_partition
					subject_contract_has_flag = succession_rights_grant_recognition
					subject_contract_has_flag = succession_rights_court_supervision
					subject_contract_has_flag = succession_rights_governor_appointment
					trigger_if = {
						limit = { NOT = { government_has_flag = government_is_feudal_admin } }
						government_allows = administrative
					}
				}
				count >= 15
			}
			any_in_list = {
				variable = list_valid_vassal
				OR = {
					subject_contract_has_flag = succession_rights_hereditary
					subject_contract_has_flag = locked_to_partition
					subject_contract_has_flag = succession_rights_grant_recognition
					subject_contract_has_flag = succession_rights_court_supervision
					subject_contract_has_flag = succession_rights_governor_appointment
					trigger_if = {
						limit = { NOT = { government_has_flag = government_is_feudal_admin } }
						government_allows = administrative
					}
				}
				percent >= 0.5
			}
		}
	}
}
