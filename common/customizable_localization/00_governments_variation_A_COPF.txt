# 风味化：封建制
GetFeudalGovName = {
	type = character

	text = {
		trigger = {
			government_has_flag = government_is_feudal
			OR = {
				culture = { has_cultural_pillar = heritage_frankish }
				liege ?= { culture = { has_cultural_pillar = heritage_frankish } }
			}
		}
		localization_key = FEUDAL_GOV_FRANCE
	}
	text = {
		trigger = {
			government_has_flag = government_is_feudal
		}
		localization_key = FEUDAL_GOV_DEFAULT
		fallback = yes
	}
}
