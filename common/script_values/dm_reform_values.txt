dm_reform_law_interest_value = {
	value = scope:story.var:dm_reform_interest_delta
	multiply = dm_reform_interest_coefficient_value
	# Movement leaders keep their dedicated table. Powerful vassals keep the
	# full generic coefficient. Pure councillors and court-position holders
	# carry one half of the corresponding powerful-vassal institutional weight.
	if = {
		limit = {
			NOT = { exists = var:movement_member }
			NOT = { is_powerful_vassal_of = scope:dm_reform_ruler }
		}
		multiply = 0.5
	}
	if = {
		limit = { var:movement_member = flag:pro_hegemon }
		add = 30
	}
	min = -100
	max = 100
}

dm_reform_stance_score_value = {
	value = dm_reform_law_interest_value
	multiply = 0.4
	if = { limit = { has_trait = diligent } add = 3 }
	if = { limit = { has_trait = lazy } add = -4 }
	if = { limit = { has_trait = brave } add = 2 }
	if = { limit = { has_trait = craven } add = -2 }
	if = { limit = { has_trait = fickle } add = 2 }
	if = { limit = { has_trait = stubborn } add = -3 }
	if = { limit = { has_trait = content } add = -3 }
	if = { limit = { has_trait = patient } add = 1 }
	if = { limit = { has_trait = paranoid } add = -2 }
	if = { limit = { has_trait = trusting } add = 1 }
	if = {
		limit = {
			has_trait = impatient
			scope:story = { var:dm_reform_progress < 50 }
		}
		add = -2
	}
	if = {
		limit = { has_trait = ambitious }
		if = {
			limit = { dm_reform_law_interest_value > 0 }
			add = 3
		}
		else_if = {
			limit = { dm_reform_law_interest_value < 0 }
			add = -4
		}
		else = { add = 1 }
	}
	if = {
		limit = {
			save_temporary_opinion_value_as = {
				name = dm_reform_ruler_opinion
				target = scope:dm_reform_ruler
			}
		}
		add = {
			value = scope:dm_reform_ruler_opinion
			if = {
				limit = { exists = scope:dm_reform_reformer }
				multiply = 0.25
			}
			else = { multiply = 0.4 }
		}
	}
	if = {
		limit = {
			exists = scope:dm_reform_reformer
			save_temporary_opinion_value_as = {
				name = dm_reform_reformer_opinion
				target = scope:dm_reform_reformer
			}
		}
		add = {
			value = scope:dm_reform_reformer_opinion
			multiply = 0.15
		}
	}
	if = {
		limit = {
			scope:story = { var:dm_reform_axis = flag:meritocracy }
			has_trait = just
		}
		add = 2
	}
	if = {
		limit = {
			scope:story = { var:dm_reform_axis = flag:fiscal_extraction }
			has_trait = greedy
		}
		add = -2
	}
	if = {
		limit = {
			scope:story = { var:dm_reform_axis = flag:ritual_legitimacy }
			has_trait = zealous
		}
		add = 3
	}
}

# Matching-trait options use the better effective attribute only among
# characters who actually possess one of the registered matching traits.
dm_reform_trait_check_diplomacy_value = {
	value = 0
	if = {
		limit = {
			scope:dm_reform_story.story_owner = {
				OR = {
					has_trait = gregarious
					has_trait = compassionate
					has_trait = forgiving
					has_trait = honest
				}
			}
		}
		value = dm_reform_ruler_check_diplomacy_value
	}
	if = {
		limit = {
			scope:dm_reform_story = { dm_reform_has_reformer_trigger = yes }
			scope:dm_reform_story.var:dm_reform_reformer = {
				OR = {
					has_trait = gregarious
					has_trait = compassionate
					has_trait = forgiving
					has_trait = honest
				}
			}
			OR = {
				scope:dm_reform_story.story_owner = {
					NOR = {
						has_trait = gregarious
						has_trait = compassionate
						has_trait = forgiving
						has_trait = honest
					}
				}
				dm_reform_reformer_check_diplomacy_value > dm_reform_ruler_check_diplomacy_value
			}
		}
		value = dm_reform_reformer_check_diplomacy_value

}
