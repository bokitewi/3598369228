# Courtier support is snapshotted on the court owner and refreshed every five years.
saas_courtier_amenity_cost_multiplier_value = {
	value = 1
	if = {
		limit = { has_variable = saas_courtier_amenity_surcharge_snapshot }
		add = var:saas_courtier_amenity_surcharge_snapshot
	}
}

# These are the sixteen final amenity prices referenced by code. Keeping the
# multiplier here preserves every vanilla minimum, discount, and free-cost branch.
court_fashion_acceptable_cost = {
	value = court_amenities_cost_level_1_fashion
	if = {
		limit = { culture = { has_cultural_parameter = reduces_court_fashion_costs_parameter } }
		multiply = 0.5
	}
	multiply = saas_courtier_amenity_cost_multiplier_value
}
