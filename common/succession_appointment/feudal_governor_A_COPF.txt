	allow_same_tier_candidates = no

			add = {
				value = martial
				add = diplomacy
				add = stewardship
				add = intrigue
				add = learning
				if = {
					limit = {
						scope:title.holder = { subject_contract_has_flag = candidate_preference_talent_only }
					}

				}
				else = {
					multiply = 1

			}

			if = { #是法理下属有地封臣
				limit = {
					AND = {
						is_landed = yes
						scope:title ?= {

						}

					limit = {

						desc = tt_is_landed_dejure_direct_vassal

					add = {

							subject_contract_has_flag = candidate_preference_powerful_gentry

								government_has_flag = government_is_feudal

						value = 10
						desc = tt_is_landed_direct_vassal
					}
				}

					subtract = {
						value = 5
						desc = tt_is_landed_direct_vassal
					}
				}
			}
			else_if = { #是辖区外有地领主
				limit = { is_landed = yes }

					limit = { is_governor = yes }

						value = -1000

						value = -1000
						desc = tt_is_landed_other_ruler
					}
				}
				
			}

			if = {
				limit = {
					scope:title.holder ?= {
						any_councillor ?= { this = root }

					limit = {
						scope:title.holder = { subject_contract_has_flag = candidate_preference_favor_nepotism }
					}

						value = 40
						desc = tt_is_liege_best_friend

					add = {
						value = 0
						desc = tt_is_liege_best_friend
					}
				}

			}

			# 禁止教士参选
			if = {
				limit = {
					is_clergy = yes
				}
				multiply = {
					value = 0
				}
				add = {
					value = -1000
					desc = tt_is_clergy
				}
			}
			# 禁止【最高领主】及其主要继承人参选
			if = {
				limit = { this = top_liege }
				multiply = {
					value = 0
				}
				add = {
					value = -1000
					desc = appointment_is_top_liege
				}
			}
			else_if = {
				limit = { is_primary_heir_of = top_liege }
				multiply = {
					value = 0
				}
			}
			# 禁止【领主】及其主要继承人参选
			else_if = {
				limit = { this = scope:title.holder.liege }
				multiply = {
					value = 0
				}
				add = {
					value = -1000
					desc = tt_is_liege
				}
			}
			else_if = {
				limit = { is_primary_heir_of = scope:title.holder.liege }
				multiply = {
					value = 0
				}
			}
		}
	}
}
