decision_petition_restore_succession = {
	ai_check_interval_by_tier = {
		barony = 0
		county = 45
		duchy = 45
		kingdom = 45
		empire = 45
		hegemony = 0
	}
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_knight_kneeling.dds"
	}
	decision_group_type = major

	cost = {
		prestige = standard_activity_cost
		gold = {
			value = 0
			if = {
				limit = {
					exists = scope:pay_homage_gold
					scope:pay_homage_gold = yes
					has_treasury = no
				}
				add = medium_gold_value
			}
		}
		treasury = {
			value = 0
			if = {
				limit = {
					exists = scope:pay_homage_gold
					scope:pay_homage_gold = yes
					has_treasury = yes
				}
				add = medium_gold_value
			}
		}
	}

	is_shown = {
		is_landed = yes
		primary_title.tier >= tier_county
		top_liege != this
		any_held_title = {
			has_variable = is_denied_title_succession
		}
	}

	is_valid = {
		exists = liege
		custom_tooltip = {
			text = i_have_denied_title_succession
			
			any_held_title = {
				has_variable = is_denied_title_succession
			}
		}
		
	}

	widget = {
		gui = "decision_view_widget_pay_homage"
		controller = decision_option_list_controller
		decision_to_second_step_button = "decision_petition_restore_succession_NEXT_STEP_BUTTON"

		item = { # 忠顺
			value = petition_restore_succession_submission
			is_valid = {}
			current_description = petition_restore_succession_submission_desc
			localization = decision_petition_restore_succession_option_submission
			icon = "gfx/interface/icons/icon_liege.dds"
			ai_chance = { # Always likely
				value = 50
				if = { # Generous vassals want to give gold if they can!
					limit = {
						ai_greed <= -50
						gold >= medium_gold_value
					}
					add = -50
				}
			}
		}

		item = { # 牵制
			value = petition_restore_succession_hook
			is_valid = {
				custom_tooltip = { # Liege does not already have a hook
					text = petition_restore_succession_hook_tt
					NOT = {
						liege ?= { has_hook = prev } 
					}
				}
			}
			current_description = petition_restore_succession_hook_desc
			localization = decision_petition_restore_succession_option_hook
			icon = "gfx/interface/icons/message_feed/hook.dds"
			ai_chance = { # More likely if family
				value = 25
				if = {
					limit = {
						liege ?= { is_close_or_extended_family_of = prev }
					}
					add = 25
				}
				if = { # Certain AI's really do not want to be indebted
					limit = {
						OR = {
							has_trait = paranoid
							has_trait = arrogant
							ai_greed >= 50
						}
					}
					add = -50
				}
			}
		}

		item = { # 礼物
			value = petition_restore_succession_gift
			# Can afford gold homage
			is_valid = { gold >= medium_gold_value }
			current_description = petition_restore_succession_gift_desc
			localization = decision_petition_restore_succession_option_gift
			icon = "gfx/interface/icons/message_feed/money.dds"
			ai_chance = { # More likely if rich
				value = 50
				if = {
					limit = { short_term_gold > major_gold_value }
					add = 50
				}
				
				if = { # Certain AI's really do not want to part with gold
					limit = {
						ai_greed >= 50
					}
					add = -100
				}
			}
		}
	}

	effect = {
		custom_tooltip = decision_petition_restore_succession_effects
		if = { # 礼物
			limit = { scope:petition_restore_succession_gift = yes }
			set_variable = {
				name = petition_restore_succession_type
				value = flag:petition_restore_succession_gift
			}
			set_variable = {
				name = petition_restore_succession_gift_value 
				value = medium_gold_value
			}
			custom_tooltip = decision_petition_restore_succession_effects_gift
		}
		else_if = { # 牵制
			limit = { scope:petition_restore_succession_hook = yes }
			set_variable = {
				name = petition_restore_succession_type
				value = flag:petition_restore_succession_hook
			}
			custom_tooltip = decision_petition_restore_succession_effects_hook
		}
		else = { # # 忠顺
			set_variable = {
				name = petition_restore_succession_type
				value = flag:petition_restore_succession_submission
			}
		}

		show_as_tooltip = {
			switch = {
				trigger = has_trait
				shy = { add_stress = minor_stress_impact_gain }
				arrogant = { add_stress = minor_stress_impact_gain }
			}
		}
		save_scope_as = scope_petition_restore_succession_vassal
		if = {
			limit = { exists = liege } # To stop errors
			liege = { save_scope_as = scope_petition_restore_succession_liege }
		}
		if = {
			limit = {
				liege = {
					has_realm_law = uncrowned
				}
			}
			custom_tooltip = petition_restore_succession_uncrowned_tt
		}
		
		scope:scope_petition_restore_succession_liege = {
			trigger_event = {
				id = petition_restore_succession.1
				delayed = yes
			}
		}
	}
	
	ai_potential = {
		is_at_war = no
	}

	ai_will_do = {
		base = -1000 #暂时停用，之后研究
		modifier = {
			add = 50
			prestige > standard_activity_cost
		}
		modifier = { # AI's who can give gold are more likely
			add = 50
			gold >= medium_gold_value
			ai_greed < 50
			primary_title.tier > tier_county
		}
		modifier = {
			add = -50
			has_relation_rival = liege
		}
		modifier = {
			add = -25
			opinion = {
				target = liege
				value < 0
			}
		}
		modifier = { # Dukes/kings should be more weighted
			add = 25
			primary_title.tier >= tier_duchy
		}
		modifier = { # Barons should be less weighted
			add = -25
			primary_title.tier < tier_county
		}
		#Unwilling to do it to lieges that are uncrowned
		#Makes it less likely vassals will be traveling during coronation
		modifier = {
			factor = 0
			liege = {
				has_realm_law = uncrowned
			}
		}
	}
}
