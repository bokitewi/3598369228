on_combat_end_winner = {
	on_actions = {
		rp_on_combat_end_winner
		rp_on_combat_end_winner_player_subject
		rp_on_combat_end_winner_campaign_incidents
	}
}

rp_on_combat_end_winner = {
	trigger = {
		exists = combat.combat_war
		OR = {
			combat.combat_war = {
				primary_attacker = { rp_is_merit_system_enabled_trigger = yes }
			}
			combat.combat_war = {
				primary_defender = { rp_is_merit_system_enabled_trigger = yes }
			}
		}
	}
	events = {
		rp_merit.0001
	}
}

rp_on_combat_end_winner_player_subject = {
	trigger = {
		exists = combat.combat_war
		side_primary_participant = {
			is_ai = yes
			is_ruler = yes
		}
		OR = {
			side_commander = { is_ai = no }
			any_side_knight = { is_ai = no }
		}
	}
	events = {
		rp_merit.1300
	}
}

	on_actions = {
		rp_on_siege_completion
		rp_on_siege_completion_campaign_incidents

rp_on_siege_completion = {
	trigger = {
		exists = root
		exists = scope:war
		root = { is_ruler = yes }
		OR = {
			scope:war = {
				primary_attacker = {
					rp_is_merit_system_enabled_trigger = yes
					OR = {
						this = root
						this = root.top_liege
					}
				}
			}
			scope:war = {
				primary_defender = {
					rp_is_merit_system_enabled_trigger = yes
					OR = {
						this = root
						this = root.top_liege
					}
				}
			}
		}
	}
	events = {
		rp_merit.0002
	}
}

on_war_won_attacker = {
	on_actions = {
		rp_on_war_won_attacker
	}
}

rp_on_war_won_attacker = {
	trigger = {
		exists = scope:attacker
		scope:attacker = { rp_is_merit_system_enabled_trigger = yes }
	}
	effect = {
		scope:attacker = {
			rp_war_end_service_merit_effect = yes
			if = {
				limit = { rp_has_rewardable_merit_subject_trigger = yes }
				set_variable = { name = rp_petition_chain_count value = 0 }
				trigger_event = {
					id = rp_merit.1000
					days = 3
				}
			}
		}
	}
}

rp_yearly_founding_stories = {
	trigger = {
		root = {
			rp_is_merit_system_enabled_trigger = yes
			NOT = { has_character_flag = rp_founding_story_cooldown }
		}
	}
	effect = {
		root = { rp_founding_story_dispatch_effect = yes }
	}
}

rp_yearly_power_overshadow_story = {
	trigger = {
		root = {
			rp_is_merit_system_enabled_trigger = yes
			NOT = { has_character_flag = rp_power_story_start_cooldown }
		}
	}
	effect = {
		root = {
			random = {
				chance = 35
				rp_try_start_power_overshadow_story_effect = yes
			}
		}
	}
}

on_title_gain = {
	on_actions = {
		rp_on_title_gain_founding_followers
	}
}

rp_on_title_gain_founding_followers = {
	trigger = {
		root = {
			is_ruler = yes
		}
	}
	effect = {
		if = {
			limit = {
				root = {
					rp_is_merit_system_enabled_trigger = yes
					highest_held_title_tier <= tier_duchy
					NOT = { has_character_flag = rp_recorded_founding_followers }
				}
				scope:title.tier >= tier_duchy
			}
			root = { rp_record_founding_followers_effect = yes }
		}
		root = {
			if = {
				limit = {
					has_character_flag = rp_founding_follower
					scope:title.tier >= tier_county
				}
				rp_establish_founding_house_effect = yes
			}
		}
	}
}

	on_actions = {
		rp_on_death_clear_merit
	}

rp_on_death_clear_merit = {
	trigger = {
		root = {
			OR = {
				has_variable = rp_merit_score
				has_character_modifier = rp_danshu_tiequan_modifier
			}
		}
	}
	effect = {
		root = {
			rp_inherit_danshu_tiequan_effect = yes
			rp_clear_merit_effect = yes
		}
	}
}
