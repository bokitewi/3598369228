saas_catalyst_alliance_added = {
	effect = {
		if = {
			limit = {
				scope:first = { saas_eligible_participant_trigger = yes saas_is_lijiao_ruler_trigger = yes }
				scope:second = { saas_eligible_participant_trigger = yes saas_is_lijiao_ruler_trigger = yes }
			}
			saas_trigger_catalyst_effect = { CATALYST = saas_catalyst_lijiao_alliance CHARACTER = scope:first }
		}
		if = {
			limit = {
				scope:first = { saas_eligible_participant_trigger = yes }
				scope:second = { saas_eligible_participant_trigger = yes }
				scope:first.faith != scope:second.faith
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_formed_interreligious_alliance_with_important_character
				CHARACTER = scope:first
			}
		}
	}
}

saas_catalyst_marriage = {
	effect = {
		if = {
			limit = {
				root = { religion = religion:lijiao_religion }
				scope:spouse = { religion = religion:lijiao_religion }
				root.top_liege = { saas_eligible_participant_trigger = yes }
				scope:spouse.top_liege = { saas_eligible_participant_trigger = yes }
				root.top_liege != scope:spouse.top_liege
			}
			saas_trigger_catalyst_effect = { CATALYST = saas_catalyst_lijiao_alliance CHARACTER = root.top_liege }
		}
	}
}

saas_catalyst_title_usurped = {
	effect = {
		if = {
			limit = { root = { saas_eligible_participant_trigger = yes } scope:transfer_type = flag:usurped }
			saas_trigger_catalyst_effect = { CATALYST = saas_catalyst_title_usurped CHARACTER = root }
		}
		if = {
			limit = {
				root = { saas_eligible_participant_trigger = yes }
				scope:title.empire ?= title:e_huaxia
				scope:transfer_type = flag:usurped
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_usurp_title_fp2
				CHARACTER = root
			}
		}
	}
}

saas_catalyst_title_created = {
	effect = {
		if = {
			limit = {
				NOT = { exists = scope:previous_holder }
				scope:title.tier >= tier_duchy
				root = { saas_is_lijiao_ruler_trigger = yes }
			}
			saas_trigger_catalyst_effect = { CATALYST = saas_catalyst_house_restored CHARACTER = root }
		}
	}
}

saas_catalyst_ruler_released = {
	effect = {
		if = {
			limit = { root = { saas_is_lijiao_ruler_trigger = yes highest_held_title_tier >= tier_kingdom } }
			saas_trigger_catalyst_effect = { CATALYST = saas_catalyst_peace_restored CHARACTER = root }
		}
		if = {
			limit = {
				root = { saas_eligible_participant_trigger = yes }
				scope:imprisoner = { saas_eligible_participant_trigger = yes }
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_release_important
				CHARACTER = root
			}
		}
	}
}

saas_catalyst_tianzi_killed = {
	effect = {
		if = {
			limit = { root = { has_title = title:e_huaxia } exists = scope:killer }
			saas_trigger_catalyst_effect = { CATALYST = saas_catalyst_tianzi_defeat CHARACTER = scope:killer }
		}
		if = {
			limit = {
				exists = scope:killer
				scope:killer = { saas_is_badao_trigger = yes }
				root = { is_ruler = yes saas_eligible_participant_trigger = yes }
			}
			saas_trigger_catalyst_effect = { CATALYST = saas_catalyst_title_usurped CHARACTER = scope:killer }
		}
		if = {
			limit = {
				exists = scope:killer
				root = { saas_eligible_participant_trigger = yes }
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_unnatural_death_important_character
				CHARACTER = root
			}
		}
	}
}

saas_catalyst_attacker_won = {
	effect = {
		if = {
			limit = { scope:attacker = { saas_is_siyi_trigger = yes } scope:defender = { saas_is_lijiao_ruler_trigger = yes } }
			saas_trigger_catalyst_effect = { CATALYST = saas_catalyst_siyi_conquest CHARACTER = scope:attacker }
		}
		if = {
			limit = {
				root.war = {
					OR = {
						using_cb = ep3_laamp_adventurer_county_invasion
						using_cb = ep3_laamp_adventurer_duchy_invasion
						using_cb = ep3_laamp_adventurer_kingdom_invasion
						using_cb = ep3_laamp_adventurer_empire_invasion
					}
				}
			}
			saas_trigger_catalyst_effect = { CATALYST = saas_catalyst_adventurer_conquest CHARACTER = scope:attacker }
		}
		if = {
			limit = {
				scope:attacker = { saas_eligible_participant_trigger = yes }
				scope:defender = { saas_eligible_participant_trigger = yes }
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_win_any_war_within_the_region
				CHARACTER = scope:attacker
			}
		}
	}
}

# The Iberian Struggle and situations use separate catalyst APIs. These child
# on_actions mirror the original generic hooks without replacing vanilla effects.

saas_fp2_catalyst_passing_time = {
	effect = {
		situation:spring_and_autumn_situation ?= {
			if = {
				limit = { situation_top_has_catalyst = catalyst_passing_of_time }
				trigger_situation_catalyst = catalyst_passing_of_time
			}
		}
	}
}

saas_fp2_catalyst_claim_gained = {
	effect = {
		if = {
			limit = {
				NOT = { scope:transfer_type = flag:inheritance }
				scope:title.empire ?= title:e_huaxia
				root = { saas_eligible_participant_trigger = yes }
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_gain_claim_on_title_fp2
				CHARACTER = root
			}
		}
	}
}

saas_fp2_catalyst_building_completed = {
	effect = {
		if = {
			limit = {
				root = { has_holding_type = castle_holding }
				root.county.holder.top_liege = { saas_eligible_participant_trigger = yes }
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_new_building_in_castle
				CHARACTER = root.county.holder.top_liege
			}
		}
		else_if = {
			limit = {
				root = { has_holding_type = city_holding }
				root.county.holder.top_liege = { saas_eligible_participant_trigger = yes }
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_new_building_in_city
				CHARACTER = root.county.holder.top_liege
			}
		}
	}
}

saas_fp2_catalyst_county_faith_changed = {
	effect = {
		if = {
			limit = {
				root.holder.top_liege = { saas_eligible_participant_trigger = yes }
				scope:old_faith != root.faith
			}

				CHARACTER = root.holder.top_liege
			}

saas_fp2_catalyst_faith_conversion = {
	effect = {
		if = {
			limit = {
				root = { saas_eligible_participant_trigger = yes }
				exists = root.capital_province
				root.capital_province.faith = root.faith
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_convert_local_culture_faith
				CHARACTER = root
			}
		}
		else_if = {
			limit = {
				root = { saas_eligible_participant_trigger = yes }
				exists = scope:actor
				scope:actor != root
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_forced_conversion
				CHARACTER = scope:actor
			}
		}
	}
}

saas_fp2_catalyst_granted_independence = {
	effect = {
		if = {
			limit = {
				root = {
					is_independent_ruler = yes
					saas_eligible_participant_trigger = yes
					has_opinion_modifier = {
						target = scope:old_liege
						modifier = granted_independence_opinion
					}
				}
				scope:old_liege = {
					saas_eligible_participant_trigger = yes
					NOT = {
						any_held_title = {
							is_de_jure_liege_or_above_target = root.primary_title
						}
					}
				}
			}
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_independence_from_non_dejure_vassal
				CHARACTER = scope:old_liege
			}
		}
	}
}

saas_fp2_catalyst_feast_guests = {
	effect = {
		if = {
			limit = {
				scope:activity = {
					NOT = { has_variable = saas_fp2_feast_catalyst_fired }
					activity_host = { saas_eligible_participant_trigger = yes }
					any_attending_character = {
						saas_eligible_participant_trigger = yes
						OR = {
							faith != scope:activity.activity_host.faith
							culture != scope:activity.activity_host.culture
						}
					}
				}
			}
			scope:activity = { set_variable = saas_fp2_feast_catalyst_fired }
			saas_trigger_catalyst_effect = {
				CATALYST = catalyst_invite_diff_faith_culture_to_feast
				CHARACTER = scope:activity.activity_host
			}
		}
	}
}
