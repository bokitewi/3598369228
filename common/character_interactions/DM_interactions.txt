测试齿序 = {
    category = interaction_category_diplomacy
    use_diplomatic_range = no
    ignores_pending_interaction_block = yes
    common_interaction = yes
    desc = LinXuan_GuiNv_desc
    auto_accept = yes
    is_shown = {
        scope:actor = {
            is_ai = yes
        }
    }
    on_accept = {
        scope:recipient = {
            save_scope_as = child
            if = {
                limit = {
                    scope:child = { is_female = yes }
                }
                scope:child = {
                    set_variable = {  name = child_rank value = father.DM_child_female_chixu }
                }
            }
            else = {
                scope:child = {
                    set_variable = {  name = child_rank value = father.DM_child_male_chixu }
                }
            }       
        }
    }
}
