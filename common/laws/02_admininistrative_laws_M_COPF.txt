#####################
# ADMINISTRATIVE LAWS
#####################
@admin_law_cooldown_value = 5

province_border_law = {
	default = province_border_law_0
	cumulative = no
	flag = admin_law
	flag = law_group_province_border #COPF

	# 暂定边界政策
	province_border_law_0 = {
		modifier = {
		}
		triggered_flag = {
			trigger = {
				NOR = {
					government_has_flag = government_is_feudal
					government_has_flag = government_is_feudal_admin
				}
			}
			flag = uses_imperial_bureaucracy
		}

		can_keep = {
			OR = {
				realm_law_use_imperial_bureaucracy = yes
				government_has_flag = government_is_feudal_admin #COPF
				any_vassals_are_governor_or_admin = yes #COPF
			}
			trigger_if = {
				limit = {
					top_liege != this
					NOR = { #COPF
						government_has_flag = government_is_feudal
						government_has_flag = government_is_feudal_admin
					}
				}
				liege = { has_realm_law = province_border_law_0 }
			}
		}

		can_pass = {
			trigger_if = {
				limit = {
					top_liege != this
					liege = { has_realm_law = province_border_law_0 }
					NOR = { #COPF
						government_has_flag = government_is_feudal
						government_has_flag = government_is_feudal_admin
					}
				}
				custom_tooltip = {
					text = imperial_bureaucracy_pending_law_change
					always = no
				}
			}
		}

		on_pass = {
			## Set cooldown
			set_variable = {
				name = admin_law_cooldown
				years = @admin_law_cooldown_value
			}
		}

		ai_will_do = {
			value = 1
		}
	}
	# 有限边界政策
	province_border_law_1 = {
		modifier = {
			administrative_government_vassal_opinion = -5
		}
		triggered_flag = {
			trigger = {
				NOR = {
					government_has_flag = government_is_feudal
					government_has_flag = government_is_feudal_admin
				}
			}
			flag = uses_imperial_bureaucracy
		}
		flag = admin_vassal_border_restricted

		can_keep = {
			OR = {
				realm_law_use_imperial_bureaucracy = yes
				government_has_flag = government_is_feudal_admin #COPF
				any_vassals_are_governor_or_admin = yes #COPF
			}
			trigger_if = {
				limit = {
					top_liege = this
					realm_law_use_imperial_bureaucracy = yes
				}
				custom_tooltip = {
					text = have_bureaucracy_2_or_higher
					OR = {
						has_realm_law = imperial_bureaucracy_2
						has_realm_law = imperial_bureaucracy_3
					}
				}
			}
			trigger_if = {
				limit = {
					top_liege = this
					realm_law_use_celestial_bureaucracy = yes
				}
				custom_tooltip = {
					text = have_celestial_bureaucracy_2_or_higher
					OR = {
						has_realm_law = celestial_bureaucracy_2
						has_realm_law = celestial_bureaucracy_3
					}
				}
			}
			trigger_if = {
				limit = {
					top_liege != this
					OR = { #COPF
						NOR = {
							government_has_flag = government_is_feudal
							government_has_flag = government_is_feudal_admin
						}
						trigger_if = {
							limit = { government_has_flag = government_is_feudal_admin }
							subject_contract_has_flag = succession_rights_governor_appointment
						}
					}
				}
				liege = { has_realm_law = province_border_law_1 }
			}
		}

		can_pass = {
			trigger_if = {
				limit = {
					top_liege = this
				}
				custom_tooltip = {
					text = admin_law_cooldown_desc
					NOT = { has_variable = admin_law_cooldown }
				}
			}
			trigger_if = {
				limit = {
					top_liege != this
					liege = { has_realm_law = province_border_law_1 }
					NOR = { #COPF
						government_has_flag = government_is_feudal
						government_has_flag = government_is_feudal_admin
					}
				}
				custom_tooltip = {
					text = imperial_bureaucracy_pending_law_change
					always = no
				}
			}
		}

		pass_cost = {
			prestige = {
				add = admin_law_prestige_cost
			}
		}

		on_pass = {
			## Set cooldown
			set_variable = {
				name = admin_law_cooldown
				years = @admin_law_cooldown_value
			}
		}

		ai_will_do = {
			value = 0
			if = {
				limit = {
					OR = {
						ai_has_cautious_personality = yes
						ai_has_economical_boom_personality = yes
					}
				}
				add = 2
			}
		}
	}
	# 固定边界政策
	province_border_law_2 = {
		modifier = {
			administrative_government_vassal_opinion = -10
		}
		triggered_flag = {
			trigger = {
				NOR = {
					government_has_flag = government_is_feudal
					government_has_flag = government_is_feudal_admin
				}
			}
			flag = uses_imperial_bureaucracy
		}
		flag = admin_vassal_border_limit

		can_keep = {
			OR = {
				realm_law_use_imperial_bureaucracy = yes
				government_has_flag = government_is_feudal_admin #COPF
				any_vassals_are_governor_or_admin = yes #COPF
			}
			trigger_if = {
				limit = {
					top_liege = this
					realm_law_use_imperial_bureaucracy = yes
				}
				custom_tooltip = {
					text = have_bureaucracy_3
					has_realm_law = imperial_bureaucracy_3
				}
			}
			trigger_if = {
				limit = {
					top_liege = this
					realm_law_use_celestial_bureaucracy = yes
				}
				custom_tooltip = {
					text = have_celestial_bureaucracy_3
					has_realm_law = celestial_bureaucracy_3
				}
			}
			trigger_if = {
				limit = {
					top_liege != this
					OR = { #COPF
						NOR = {
							government_has_flag = government_is_feudal
							government_has_flag = government_is_feudal_admin
						}
						trigger_if = {
							limit = { government_has_flag = government_is_feudal_admin }
							subject_contract_has_flag = succession_rights_governor_appointment
						}
					}
				}
				liege = { has_realm_law = province_border_law_2 }
			}
		}

		can_pass = {
			trigger_if = {
				limit = {
					top_liege = this
				}
				custom_tooltip = {
					text = admin_law_cooldown_desc
					NOT = { has_variable = admin_law_cooldown }
				}
			}
			trigger_if = {
				limit = {
					top_liege != this
					liege = { has_realm_law = province_border_law_2 }
					NOR = { #COPF
						government_has_flag = government_is_feudal
						government_has_flag = government_is_feudal_admin
					}
				}
				custom_tooltip = {
					text = imperial_bureaucracy_pending_law_change
					always = no
				}
			}
		}

		pass_cost = {
			prestige = {
				add = admin_law_prestige_cost
			}
		}

		on_pass = {
			## Set cooldown
			set_variable = {
				name = admin_law_cooldown
				years = @admin_law_cooldown_value
			}
		}

		ai_will_do = {
			value = 0
		}
	}
}
