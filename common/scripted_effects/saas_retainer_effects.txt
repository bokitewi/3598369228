saas_refresh_courtier_amenity_surcharge_effect = {
	remove_variable = saas_courtier_amenity_surcharge_snapshot
	if = {
		limit = { has_royal_court = yes }
		set_variable = {
			name = saas_courtier_amenity_surcharge_snapshot
			value = saas_courtier_amenity_surcharge_value
		}
	}
}

saas_accept_retainer_candidate_effect = {
	if = {
		limit = { $RETAINER$ = { NOT = { is_courtier_of = root } } }
		add_courtier = $RETAINER$
	}
	$RETAINER$ = {
		add_opinion = {
			target = root
			modifier = saas_retainer_patronage_opinion
		}
		add_character_flag = {
			flag = saas_retainer_event_cooldown
			years = 10
		}
	}
	saas_set_retainer_project_phase_effect = yes
}

saas_decline_retainer_candidate_effect = {
	$RETAINER$ = {
		add_opinion = {
			target = root
			modifier = saas_retainer_rejected_opinion
		}
		add_character_flag = {
			flag = saas_retainer_event_cooldown
			years = 10
		}
	}
	remove_character_flag = saas_retainer_project_active
}

saas_clear_retainer_project_effect = {
	remove_character_flag = saas_retainer_project_active
	remove_character_flag = saas_retainer_project_restoration_phase
	remove_character_flag = saas_retainer_project_collapse_phase
	remove_character_flag = saas_retainer_project_warring_phase
	remove_variable = saas_retainer_investment
}

saas_set_retainer_project_phase_effect = {
	remove_character_flag = saas_retainer_project_restoration_phase
	remove_character_flag = saas_retainer_project_collapse_phase
	remove_character_flag = saas_retainer_project_warring_phase
	if = {
		limit = { saas_is_ritual_restoration_phase_trigger = yes }
		add_character_flag = saas_retainer_project_restoration_phase
	}
	else_if = {
		limit = { saas_is_ritual_collapse_phase_trigger = yes }
		add_character_flag = saas_retainer_project_collapse_phase
	}
	else_if = {
		limit = { saas_is_warring_states_phase_trigger = yes }
		add_character_flag = saas_retainer_project_warring_phase
	}
}

saas_move_talent_to_patron_effect = {
	$TALENT$ = {
		add_character_flag = {
			flag = saas_recently_recruited_talent
			years = 10
		}
		every_traveling_family_member = {
			move_to_pool_at = $PATRON$.capital_province
			visit_court_of = $PATRON$
		}
	}
}

saas_compete_for_talent_effect = {
	save_scope_as = saas_migrant_talent
	random_independent_ruler = {
		limit = {
			saas_recruitment_target_trigger = yes
			has_character_modifier = saas_recruit_talent_edict_modifier
			trigger_if = {
				limit = { exists = scope:saas_migrant_talent.host }
				this != scope:saas_migrant_talent.host
			}
		}
		weight = {
			base = 100
			modifier = { add = 25 highest_held_title_tier = tier_duchy }
			modifier = { add = 50 highest_held_title_tier = tier_kingdom }
			modifier = { add = 80 highest_held_title_tier >= tier_empire }
			modifier = { add = 25 realm_size >= 20 }
			modifier = { add = 25 realm_size >= 50 }
			modifier = { add = 25 realm_size >= 100 }
			modifier = { add = 10 max_military_strength >= 1000 }
			modifier = { add = 20 max_military_strength >= 3000 }
			modifier = { add = 30 max_military_strength >= 8000 }
			modifier = {
				add = {
					value = prestige_level
					multiply = 10
				}
			}
			modifier = {
				add = {
					value = court_grandeur_current_level
					multiply = 5
				}
				has_royal_court = yes
			}
			modifier = {
				add = 40
				culture = scope:saas_migrant_talent.culture
				scope:saas_migrant_talent = {
					OR = {
						has_trait = content
						has_trait = loyal
					}
				}
			}
			modifier = {
				add = 40
				faith = scope:saas_migrant_talent.faith
				scope:saas_migrant_talent = {
					OR = {
						has_trait = zealous
						has_trait = loyal
					}
				}
			}
			modifier = {
				add = 60
				highest_held_title_tier >= tier_kingdom
				scope:saas_migrant_talent = { has_trait = ambitious }
			}
			modifier = {
				add = 50
				max_military_strength >= 3000
				scope:saas_migrant_talent = { has_trait = ambitious }
			}
		}
		save_scope_as = saas_winning_patron
	}
	if = {
		limit = { exists = scope:saas_winning_patron }
		saas_move_talent_to_patron_effect = {
			TALENT = scope:saas_migrant_talent
			PATRON = scope:saas_winning_patron
		}
	}
}

saas_create_elite_guest_effect = {
	save_scope_as = saas_elite_patron
	create_character = {
		template = saas_elite_pool_character
		location = root.capital_province
		save_scope_as = saas_elite_guest
	}
	scope:saas_elite_guest = {
		random_list = {
			20 = {
				trigger = {
					has_dlc_feature = roads_to_power
					is_physically_able_adult = yes
					exists = location.county
					ai_can_valid_to_create_laamp_trigger = yes
				}
				add_gold = 500
				create_landless_adventurer_title_effect = {
					REASON = flag:voluntary
					FLAVOR_CHAR = scope:saas_elite_guest
				}
				scope:saas_elite_patron = {
					send_interface_message = {
						type = event_toast_effect_neutral
						title = saas_elite_adventurer_spawned_title
						left_icon = scope:saas_elite_guest
						custom_tooltip = saas_elite_adventurer_spawned_desc
					}
				}
			}
			80 = {
				visit_court_of = scope:saas_elite_patron
				scope:saas_elite_patron = {
					send_interface_message = {
						type = event_toast_effect_neutral
						title = saas_elite_guest_arrived_title
						left_icon = scope:saas_elite_guest
						custom_tooltip = saas_elite_guest_arrived_desc
					}
				}
			}
		}
	}
}
