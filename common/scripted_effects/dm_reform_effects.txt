dm_reform_install_reformer_effect = {
	# ROOT: reform story. REFORMER must be a character scope that accepted.
	set_variable = {
		name = dm_reform_reformer
		value = $REFORMER$
	}
	remove_list_variable = {
		name = dm_reform_pending_participants
		target = $REFORMER$
	}
	remove_list_variable = {
		name = dm_reform_supporters
		target = $REFORMER$
	}
	remove_list_variable = {
		name = dm_reform_opponents
		target = $REFORMER$
	}
	remove_variable = dm_reform_pending_reformer
}

dm_reform_send_counter_notification_effect = {
	# ROOT: a potential character recipient; scope:dm_reform_story is persistent.
	if = {
		limit = {
			is_ai = no
			NOT = {
				scope:dm_reform_story = {
					is_target_in_variable_list = {
						name = dm_reform_notification_recipients
						target = root
					}
				}
			}
		}
		scope:dm_reform_story = {
			add_to_variable_list = {
				name = dm_reform_notification_recipients
				target = root
			}
		}
		if = {
			limit = {
				scope:dm_reform_story = { var:dm_reform_notification_score > 0 }
			}
			send_interface_message = {
				type = event_generic_good_text
				title = dm_reform_counter_message_title
				desc = dm_reform_counter_message_desc
				left_icon = scope:dm_reform_ruler
			}
		}
		else_if = {
			limit = {
				scope:dm_reform_story = { var:dm_reform_notification_score < 0 }
			}
			send_interface_message = {
				type = event_generic_bad_text
				title = dm_reform_counter_message_title
				desc = dm_reform_counter_message_desc
				left_icon = scope:dm_reform_ruler
			}
		}
		else = {
			send_interface_message = {
				type = event_generic_neutral_text
				title = dm_reform_counter_message_title
				desc = dm_reform_counter_message_desc
				left_icon = scope:dm_reform_ruler
			}
		}
	}
}

dm_reform_apply_counter_change_effect = {
	# ROOT: reform story. All progress/support mutations use this transaction.
	save_scope_as = dm_reform_story
	story_owner = { save_scope_as = dm_reform_ruler }
	set_variable = {
		name = dm_reform_old_progress_display
		value = {
			value = var:dm_reform_progress
			floor = yes
		}
	}
	set_variable = {
		name = dm_reform_old_support_display
		value = var:dm_reform_support
	}
	change_variable = { name = dm_reform_progress add = $PROGRESS$ }
	change_variable = { name = dm_reform_support add = $SUPPORT$ }
	if = {
		limit = { var:dm_reform_progress < 0 }
		set_variable = { name = dm_reform_progress value = 0 }
	}
	else_if = {
		limit = { var:dm_reform_progress > 100 }
		set_variable = { name = dm_reform_progress value = 100 }
	}
	if = {
		limit = { var:dm_reform_support < -10 }
		set_variable = { name = dm_reform_support value = -10 }
	}
	else_if = {
		limit = { var:dm_reform_support > 10 }
		set_variable = { name = dm_reform_support value = 10 }
	}
	set_variable = {
		name = dm_reform_new_progress_display
		value = {
			value = var:dm_reform_progress
			floor = yes
		}
	}
	set_variable = {
		name = dm_reform_new_support_display
		value = var:dm_reform_support
	}
	set_variable = {
		name = dm_reform_progress_display_delta
		value = var:dm_reform_new_progress_display
	}
	change_variable = {
		name = dm_reform_progress_display_delta
		subtract = var:dm_reform_old_progress_display
	}
	set_variable = {
		name = dm_reform_support_display_delta
		value = var:dm_reform_new_support_display
	}
	change_variable = {
		name = dm_reform_support_display_delta
		subtract = var:dm_reform_old_support_display
	}
	set_variable = {
		name = dm_reform_notification_score
		value = var:dm_reform_support_display_delta
	}
	change_variable = { name = dm_reform_notification_score multiply = 3 }
	change_variable = {
		name = dm_reform_notification_score
		add = var:dm_reform_progress_display_delta
	}
	set_variable = { name = dm_reform_last_result value = flag:$RESULT$ }
	if = {
		limit = {
			OR = {
				var:dm_reform_progress_display_delta != 0
				var:dm_reform_support_display_delta != 0
			}
		}
		clear_variable_list = dm_reform_notification_recipients
		scope:dm_reform_ruler = { dm_reform_send_counter_notification_effect = yes }
		if = {
			limit = { dm_reform_has_reformer_trigger = yes }
			var:dm_reform_reformer = {
				dm_reform_send_counter_notification_effect = yes
			}
		}
		every_in_list = {
			variable = dm_reform_supporters
			dm_reform_send_counter_notification_effect = yes
		}
		every_in_list = {
			variable = dm_reform_opponents
			dm_reform_send_counter_notification_effect = yes
		}
		every_in_list = {
			variable = dm_reform_pending_participants
			dm_reform_send_counter_notification_effect = yes
		}
		clear_variable_list = dm_reform_notification_recipients
	}
}

dm_reform_ai_quick_resolve_effect = {
	# ROOT: reform story. Pure-AI path: no visible event, actor, chain or cooldown.
	if = {
		limit = { var:dm_reform_ai_check_value <= $TIER_1$ }
		random_list = {
			35 = { set_variable = { name = dm_reform_ai_result value = flag:disaster } }
			45 = { set_variable = { name = dm_reform_ai_result value = flag:failure } }
			19 = { set_variable = { name = dm_reform_ai_result value = flag:success } }
			1 = { set_variable = { name = dm_reform_ai_result value = flag:great_success } }
		}
	}
	else_if = {
		limit = { var:dm_reform_ai_check_value <= $TIER_2$ }
		random_list = {
			25 = { set_variable = { name = dm_reform_ai_result value = flag:disaster } }
			40 = { set_variable = { name = dm_reform_ai_result value = flag:failure } }
			30 = { set_variable = { name = dm_reform_ai_result value = flag:success } }
			5 = { set_variable = { name = dm_reform_ai_result value = flag:great_success } }
		}
	}
	else_if = {
		limit = { var:dm_reform_ai_check_value <= $TIER_3$ }
		random_list = {
			15 = { set_variable = { name = dm_reform_ai_result value = flag:disaster } }
			35 = { set_variable = { name = dm_reform_ai_result value = flag:failure } }
			40 = { set_variable = { name = dm_reform_ai_result value = flag:success } }
			10 = { set_variable = { name = dm_reform_ai_result value = flag:great_success } }
		}
	}
	else_if = {
		limit = { var:dm_reform_ai_check_value <= $TIER_4$ }
		random_list = {
			10 = { set_variable = { name = dm_reform_ai_result value = flag:disaster } }
			25 = { set_variable = { name = dm_reform_ai_result value = flag:failure } }
			45 = { set_variable = { name = dm_reform_ai_result value = flag:success } }
			20 = { set_variable = { name = dm_reform_ai_result value = flag:great_success } }
		}
	}
	else_if = {
		limit = { var:dm_reform_ai_check_value <= $TIER_5$ }
		random_list = {
			5 = { set_variable = { name = dm_reform_ai_result value = flag:disaster } }
			15 = { set_variable = { name = dm_reform_ai_result value = flag:failure } }
			50 = { set_variable = { name = dm_reform_ai_result value = flag:success } }
			30 = { set_variable = { name = dm_reform_ai_result value = flag:great_success } }
		}
	}
	else = {
		random_list = {
			1 = { set_variable = { name = dm_reform_ai_result value = flag:disaster } }
			9 = { set_variable = { name = dm_reform_ai_result value = flag:failure } }
			50 = { set_variable = { name = dm_reform_ai_result value = flag:success } }
			40 = { set_variable = { name = dm_reform_ai_result value = flag:great_success } }
		}
	}
	if = {

		dm_reform_apply_counter_change_effect = {
			PROGRESS = $DISASTER_PROGRESS$
			SUPPORT = $DISASTER_SUPPORT$
			RESULT = disaster
		}

	else_if = {

			RESULT = great_success
		}

	remove_variable = dm_reform_ai_check_value

	# ROOT: reform story. Reuse the real independence faction system and real opponents.
	story_owner = { save_scope_as = dm_reform_ruler }
	ordered_in_list = {
		variable = dm_reform_opponents
		limit = {
			is_landed = yes
			can_create_faction = {
				type = independence_faction
				target = scope:dm_reform_ruler
			}
		}
		order_by = current_military_strength
		position = 0
		save_scope_as = dm_reform_rebel_leader
	}
	scope:dm_reform_rebel_leader ?= {
		create_faction = {
			type = independence_faction
			target = scope:dm_reform_ruler
		}
		joined_faction ?= { save_scope_as = dm_reform_rebel_faction }
	}
	if = {
		limit = { exists = scope:dm_reform_rebel_faction }
		every_in_list = {
			variable = dm_reform_opponents
			limit = {
				is_landed = yes
				NOT = { this = scope:dm_reform_rebel_leader }
				can_join_faction = scope:dm_reform_rebel_faction
			}
			join_faction = scope:dm_reform_rebel_faction
		}
		scope:dm_reform_rebel_faction = {
			add_faction_discontent = 100
			faction_start_war = {}
		}
	}

dm_reform_check_immediate_outcome_effect = {
	if = {
		limit = { var:dm_reform_progress < 0 }
		set_variable = { name = dm_reform_progress value = 0 }
	}
	else_if = {
		limit = { var:dm_reform_progress > 100 }
		set_variable = { name = dm_reform_progress value = 100 }
	}
	if = {
		limit = { var:dm_reform_support < -10 }
		set_variable = { name = dm_reform_support value = -10 }
	}
	else_if = {
		limit = { var:dm_reform_support > 10 }
		set_variable = { name = dm_reform_support value = 10 }
	}
	if = {
		limit = { var:dm_reform_support <= -10 }
		dm_reform_extreme_failure_effect = yes
	}
	else_if = {
		limit = { var:dm_reform_progress >= 100 }
		dm_reform_success_effect = yes
	}
}
