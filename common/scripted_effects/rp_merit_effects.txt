rp_add_merit_effect = {
	save_scope_as = rp_merit_recipient

	if = {
		limit = { exists = $LIEGE$ }
		set_variable = {
			name = rp_merit_owner
			value = $LIEGE$
		}
		$LIEGE$ = {
			if = {
				limit = {
					NOT = {
						is_target_in_variable_list = {
							name = rp_merit_roster
							target = scope:rp_merit_recipient
						}
					}
				}
				add_to_variable_list = {
					name = rp_merit_roster
					target = scope:rp_merit_recipient
				}
			}
		}
	}

	if = {
		limit = { NOT = { has_variable = rp_merit_score } }
		set_variable = {
			name = rp_merit_score
			value = 0
		}
	}
	change_variable = {
		name = rp_merit_score
		add = $VALUE$
	}
	if = {
		limit = { var:rp_merit_score > 200 }
		set_variable = {
			name = rp_merit_score
			value = 200
		}
	}
	if = {
		limit = { var:rp_merit_score < -100 }
		set_variable = {
			name = rp_merit_score
			value = -100
		}
	}
	if = {
		limit = {
			var:rp_merit_score > 0
			OR = {
				NOT = { has_variable = rp_merit_peak_score }
				var:rp_merit_score > var:rp_merit_peak_score
			}
		}
		set_variable = {
			name = rp_merit_peak_score
			value = var:rp_merit_score
		}
	}
	rp_update_founding_follower_status_effect = yes
	rp_update_great_house_status_effect = yes
}

rp_consume_merit_effect = {
	if = {
		limit = { NOT = { has_variable = rp_merit_score } }
		set_variable = {
			name = rp_merit_score
			value = 0
		}
	}
	change_variable = {
		name = rp_merit_score
		add = -$VALUE$
	}
	if = {
		limit = { var:rp_merit_score < 0 }
		set_variable = {
			name = rp_merit_score
			value = 0
		}
	}
}

rp_update_founding_follower_status_effect = {
	if = {
		limit = { has_character_flag = rp_founding_follower }
		if = {
			limit = {
				OR = {
					AND = {
						has_variable = rp_founding_seniority
						var:rp_founding_seniority >= 40
					}
					AND = {
						has_variable = rp_merit_peak_score
						var:rp_merit_peak_score >= 150
					}
				}
			}
			if = {
				limit = { has_trait = rp_founding_follower_trait }
				remove_trait = rp_founding_follower_trait
			}
			if = {
				limit = { has_trait = rp_core_follower_trait }
				remove_trait = rp_core_follower_trait
			}
			if = {
				limit = { NOT = { has_trait = rp_pillar_follower_trait } }
				add_trait = rp_pillar_follower_trait
			}
		}
		else_if = {
			limit = {
				OR = {
					AND = {
						has_variable = rp_founding_seniority
						var:rp_founding_seniority >= 20
					}
					AND = {
						has_variable = rp_merit_peak_score
						var:rp_merit_peak_score >= 80
					}
				}
			}
			if = {
				limit = { has_trait = rp_founding_follower_trait }
				remove_trait = rp_founding_follower_trait
			}
			if = {
				limit = { has_trait = rp_pillar_follower_trait }
				remove_trait = rp_pillar_follower_trait
			}
			if = {
				limit = { NOT = { has_trait = rp_core_follower_trait } }
				add_trait = rp_core_follower_trait
			}
		}
		else = {
			if = {
				limit = { has_trait = rp_core_follower_trait }
				remove_trait = rp_core_follower_trait
			}
			if = {
				limit = { has_trait = rp_pillar_follower_trait }
				remove_trait = rp_pillar_follower_trait
			}
			if = {
				limit = { NOT = { has_trait = rp_founding_follower_trait } }
				add_trait = rp_founding_follower_trait
			}
		}
		if = {
			limit = { highest_held_title_tier >= tier_county }
			rp_establish_founding_house_effect = yes
		}
	}
}

				OR = {
					has_government = republic_government
					has_government = theocracy_government
					has_government = mercenary_government
					has_government = holy_order_government
					has_government = landless_adventurer_government
					has_government = nomad_government
					has_government = herder_government
				}
			}
		}
		if = {
			limit = { government_has_flag = government_is_clan }
			scope:rp_petitioner = { change_government = clan_government }
		}
		else_if = {
			limit = { government_has_flag = government_is_tribal }
			scope:rp_petitioner = { change_government = tribal_government }
		}
		else = {
			scope:rp_petitioner = { change_government = feudal_admin_government }
		}
	}
}

rp_apply_defection_effect = {
	scope:rp_campaign_actor = {
		add_character_modifier = {
			modifier = rp_traitor_modifier
			years = 10
		}
		add_opinion = {
			target = root
			modifier = rp_merit_refused_harsh_opinion
		}
		rp_clear_merit_effect = yes
	}
	add_prestige = -150
	add_character_modifier = {
		modifier = rp_campaign_disorder_modifier
		years = 3
	}
	if = {
		limit = {
			exists = $WAR$
			scope:rp_campaign_actor = { is_landed = yes }
			$WAR$ = { primary_attacker = { this = root } }
		}
		$WAR$ = {
			remove_participant = scope:rp_campaign_actor
			add_defender = scope:rp_campaign_actor
		}
	}
	else_if = {
		limit = {
			exists = $WAR$
			scope:rp_campaign_actor = { is_landed = yes }
			$WAR$ = { primary_defender = { this = root } }
		}
		$WAR$ = {
			remove_participant = scope:rp_campaign_actor
			add_attacker = scope:rp_campaign_actor
		}
	}
	if = {
		limit = {
			scope:rp_campaign_actor = {
				is_landed = yes
				is_vassal_of = root
				can_create_faction = {
					type = independence_faction
					target = root
				}
			}
		}
		scope:rp_campaign_actor = {
			create_faction = {
				type = independence_faction
				target = root
			}
			add_character_modifier = {
				modifier = rp_power_rebel_pressure_modifier
				years = 5
			}
		}
	}
	if = {
		limit = {
			scope:rp_campaign_actor = {
				has_variable = rp_merit_peak_score
				var:rp_merit_peak_score >= 100
			}
		}
		add_character_modifier = {
			modifier = rp_campaign_disorder_modifier
			years = 5
		}
		add_dread = -10
	}
	rp_mark_campaign_actor_used_effect = yes
}
