dm_create_character_GuoHao_GongJuRen_effect = {#庙谥国号工具人
    hidden_effect = {
        create_character = {
            save_scope_as = Gongjuren
        }
        add_courtier = scope:Gongjuren
        scope:Gongjuren = { 
            give_nickname = DMGuoHao_global_var 
            death = yes
        }
        set_global_variable = {
            name = GuoHao_var_global
            value = scope:Gongjuren
            days = 1
        }
    }
}


pinfei_list_effect = {
    clear_variable_list = Gongshi_pinfei_list
    ordered_consort = {
        order_by = {
            value = 0
            if = { limit = { this = root.primary_spouse } add = 99 }
        }
        max = 100
        root = {
            add_to_variable_list = {
                name = Gongshi_pinfei_list
                target = prev
            }
        }
    }
}

    }
}

LiGuo_years_effect = {
    primary_title = {
        ordered_past_holder = {
            limit = {
                dynasty ?= scope:actor.dynasty
            }
            order_by = days_since_death
            max = 1
            save_scope_as = kaiguoxianzu
        }
    }
    set_variable = {
        name = GuoJia_years 
        value = {
            value = scope:kaiguoxianzu.days_since_death 
            divide = 365
            ceiling = yes
            min = 0
        }
    }
    
}

GuoJun_var_effect = {
    set_variable = { name = GuoJun_var value = 0 } 
    primary_title = {
        every_past_holder = {
            even_if_dead = yes
            limit = {
                dynasty ?= scope:actor.dynasty
            }
            change_variable = { name = GuoJun_var add = 1 }
        }
    }
}
