# Stewardship retainer proposes a commercial investment.
saas_retainer.1100 = {
	type = character_event
	title = saas_retainer.1100.t
	desc = saas_retainer.1100.desc
	theme = stewardship
	left_portrait = {
		character = scope:retainer
		animation = personality_rational
	}

	trigger = {
		is_landed = yes
		NOT = { has_character_flag = saas_retainer_project_active }
		any_courtier_or_guest = { saas_retainer_stewardship_candidate_trigger = yes }
	}

	immediate = {
		add_character_flag = {
			flag = saas_retainer_project_active
			years = 2
		}
		random_courtier_or_guest = {
			limit = { saas_retainer_stewardship_candidate_trigger = yes }
			weight = {
				base = 1
				modifier = { add = stewardship }
			}
			save_scope_as = retainer
		}
	}

	option = {
		name = saas_retainer.1100.a
		trigger = { gold >= 10 }
		saas_accept_retainer_candidate_effect = { RETAINER = scope:retainer }
		pay_short_term_gold = { target = scope:retainer gold = 10 }
		set_variable = { name = saas_retainer_investment value = 10 }
		trigger_event = { id = saas_retainer.1101 months = { 4 8 } }
		ai_chance = { base = 50 modifier = { add = 25 gold >= 100 } }
	}

	option = {
		name = saas_retainer.1100.b
		trigger = { gold >= 25 }
		saas_accept_retainer_candidate_effect = { RETAINER = scope:retainer }
		pay_short_term_gold = { target = scope:retainer gold = 25 }
		set_variable = { name = saas_retainer_investment value = 25 }
		trigger_event = { id = saas_retainer.1101 months = { 4 8 } }
		ai_chance = { base = 35 modifier = { add = 30 gold >= 250 } }
	}

	option = {
		name = saas_retainer.1100.c
		trigger = { gold >= 50 }
		saas_accept_retainer_candidate_effect = { RETAINER = scope:retainer }
		pay_short_term_gold = { target = scope:retainer gold = 50 }
		set_variable = { name = saas_retainer_investment value = 50 }
		trigger_event = { id = saas_retainer.1101 months = { 4 8 } }
		ai_chance = { base = 15 modifier = { add = 50 gold >= 500 } }
	}

	option = {
		name = saas_retainer.1100.d
		saas_decline_retainer_candidate_effect = { RETAINER = scope:retainer }
		ai_chance = { base = 25 modifier = { add = 100 gold < 10 } }
	}
}

saas_retainer.1101 = {
	type = character_event
	title = saas_retainer.1101.t
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = { has_character_flag = saas_retainer_project_success }
				desc = saas_retainer.1101.desc.success
			}
			triggered_desc = {
				trigger = { has_character_flag = saas_retainer_project_invalid }
				desc = saas_retainer.1101.desc.invalid
			}
			desc = saas_retainer.1101.desc.failure
		}
	}
	theme = stewardship
	left_portrait = { character = scope:retainer animation = personality_rational }

	immediate = {
		if = {
			limit = {
				is_alive = yes
				is_landed = yes
				saas_retainer_project_phase_valid_trigger = yes
				scope:retainer ?= {
					is_alive = yes
					is_courtier_of = root
				}
				has_variable = saas_retainer_investment
			}
			random = {
				chance = {
					value = saas_retainer_stewardship_success_value
				}
				add_character_flag = saas_retainer_project_success
			}
		}
		else = { add_character_flag = saas_retainer_project_invalid }
	}

	option = {
		name = saas_retainer.1101.a
		trigger = { has_character_flag = saas_retainer_project_success }
		if = {
			limit = { var:saas_retainer_investment = 10 }
			add_gold = 20
		}
		else_if = {
			limit = { var:saas_retainer_investment = 25 }
			add_gold = 50
		}
		else_if = {
			limit = { var:saas_retainer_investment = 50 }
			add_gold = 100
		}
		scope:retainer = {
			add_opinion = {
				target = root
				modifier = saas_retainer_project_success_opinion
			}
		}
		remove_character_flag = saas_retainer_project_success
		saas_clear_retainer_project_effect = yes
	}

	option = {
		name = saas_retainer.1101.b
		trigger = {
			NOR = {
				has_character_flag = saas_retainer_project_success
				has_character_flag = saas_retainer_project_invalid
			}
		}
		scope:retainer = {
			add_opinion = {
				target = root
				modifier = saas_retainer_project_failure_opinion
			}
		}
		saas_clear_retainer_project_effect = yes
	}

	option = {
		name = saas_retainer.1101.c
		trigger = { has_character_flag = saas_retainer_project_invalid }
		remove_character_flag = saas_retainer_project_invalid
		saas_clear_retainer_project_effect = yes
	}
}

				flag = saas_salary_request_cooldown
				years = 10
			}
		}

			base = 50
			modifier = { add = 40 has_trait = generous }

		name = saas_retainer.1600.b

			add_character_modifier = {
				modifier = saas_salary_refused_modifier
				years = 5
			}

				target = root
				modifier = saas_salary_raise_refused_opinion
