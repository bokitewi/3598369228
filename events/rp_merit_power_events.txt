namespace = rp_merit

rp_merit.4000 = {
	type = character_event
	title = rp_merit.4000.t
	theme = intrigue
	override_background = { reference = throne_room_east }
	desc = rp_merit.4000.desc
	right_portrait = scope:story.var:rp_power_hero

	immediate = {
		scope:story.var:rp_power_hero = { save_scope_as = rp_power_hero }
	}

	trigger = {
		exists = scope:story
		scope:story = { rp_power_story_hero_valid_trigger = yes }
	}

	option = {
		name = rp_merit.4000.a
		trigger = { gold >= 80 }
		remove_short_term_gold = 80
		add_prestige = 50
		scope:story.var:rp_power_hero = {
			add_gold = 80
			add_opinion = { target = root modifier = rp_merit_reward_gold_opinion }
		}
		scope:story = {
			change_variable = { name = rp_power_suspicion add = 10 }
			set_variable = { name = rp_power_phase value = 2 }
		}
	}

	option = {
		name = rp_merit.4000.b
		scope:story.var:rp_power_hero = {
			add_opinion = { target = root modifier = rp_merit_refused_harsh_opinion }
			add_character_modifier = { modifier = rp_power_stripped_modifier years = 5 }
			set_variable = { name = rp_power_awareness value = 25 }
		}
		scope:story = {
			change_variable = { name = rp_power_suspicion add = 30 }
			set_variable = { name = rp_power_phase value = 2 }
		}
	}

	option = {
		name = rp_merit.4000.c
		scope:story.var:rp_power_hero = {
			add_character_modifier = { modifier = rp_power_under_suspicion_modifier years = 5 }
			set_variable = { name = rp_power_awareness value = 10 }
		}
		scope:story = {
			change_variable = { name = rp_power_suspicion add = 20 }
			set_variable = { name = rp_power_phase value = 2 }
		}
	}
}

rp_merit.4001 = {
	hidden = yes

	immediate = {
		if = {
			limit = {
				OR = {
					NOT = { exists = scope:story }
					NOT = { scope:story = { rp_power_story_hero_valid_trigger = yes } }
				}
			}
			scope:story ?= { end_story = yes }
		}
		else_if = {
			limit = {
				scope:story = {
					has_variable = rp_power_phase
					var:rp_power_phase >= 3
				}
			}
			trigger_event = rp_merit.4300
		}
		else_if = {
			limit = {
				scope:story = {
					has_variable = rp_power_phase
					var:rp_power_phase = 2
					OR = {
						var:rp_power_suspicion >= 60
						var:rp_power_cycle >= 3
					}
				}
			}
			scope:story = { set_variable = { name = rp_power_phase value = 3 } }
			trigger_event = rp_merit.4300
		}
		else = {
			scope:story = { change_variable = { name = rp_power_cycle add = 1 } }
			random_list = {
				34 = { trigger_event = rp_merit.4201 }
				33 = { trigger_event = rp_merit.4202 }
				33 = { trigger_event = rp_merit.4203 }
			}
		}
	}
}

rp_merit.4401 = {
	type = character_event
	title = rp_merit.4401.t
	theme = intrigue
	override_background = { reference = throne_room_east }
	desc = rp_merit.4401.desc
	right_portrait = scope:story.var:rp_power_hero

	immediate = {
		scope:story.var:rp_power_hero = { save_scope_as = rp_power_hero }
	}

	trigger = {
		exists = scope:story
		exists = scope:story.var:rp_power_hero
		scope:story.var:rp_power_hero = { is_alive = yes }
	}

	option = {
		name = rp_merit.4401.a
		if = {
			limit = { scope:story.var:rp_power_hero = { is_imprisoned = no } }
			imprison = {
				target = scope:story.var:rp_power_hero
				type = house_arrest
			}
		}
		scope:story.var:rp_power_hero = {
			death = {
				death_reason = death_execution
				killer = root
			}
		}
		add_dread = 30
		add_prestige = -150
		scope:story = { end_story = yes }
	}
}
