namespace = rp_merit

rp_merit.2000 = {
	scope = combat_side
	hidden = yes

	immediate = {
		combat.combat_war = { save_scope_as = rp_current_war }

		if = {
			limit = {
				combat.combat_war = {
					primary_attacker = { rp_is_merit_system_enabled_trigger = yes }
				}
				side_primary_participant = { is_attacker_in_war = root.combat.combat_war }
			}
			combat.combat_war.primary_attacker = { save_scope_as = rp_merit_liege }
		}
		else_if = {
			limit = {
				combat.combat_war = {
					primary_defender = { rp_is_merit_system_enabled_trigger = yes }
				}
				side_primary_participant = { is_defender_in_war = root.combat.combat_war }
			}
			combat.combat_war.primary_defender = { save_scope_as = rp_merit_liege }
		}

		side_commander ?= {
			if = {
				limit = { rp_is_high_merit_campaign_actor_trigger = { LIEGE = scope:rp_merit_liege VALUE = 30 } }
				save_scope_as = rp_campaign_actor
			}
		}
		if = {
			limit = { NOT = { exists = scope:rp_campaign_actor } }
			random_side_knight = {
				limit = { rp_is_high_merit_campaign_actor_trigger = { LIEGE = scope:rp_merit_liege VALUE = 30 } }
				save_scope_as = rp_campaign_actor
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = {
					rp_can_fire_campaign_incident_trigger = yes
					NOT = { has_character_flag = rp_yang_zhen_chain_cooldown }
				}
				scope:rp_campaign_actor = {
					OR = {
						has_trait = vengeful
						has_trait = greedy
						has_trait = arrogant
					}
				}
			}
			random = {
				chance = 6
				scope:rp_merit_liege = {
					rp_mark_campaign_incident_cooldown_effect = { DAYS = 180 }
					add_character_flag = {
						flag = rp_yang_zhen_chain_cooldown
						days = 365
					}
					trigger_event = rp_merit.2010
				}
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = {
					rp_can_fire_campaign_incident_trigger = yes
					NOT = { has_character_flag = rp_pursuit_chain_cooldown }
					rp_has_another_high_merit_actor_trigger = { VALUE = 40 ACTOR = scope:rp_campaign_actor }
				}
				scope:rp_campaign_actor = {
					has_variable = rp_merit_score
					var:rp_merit_score >= 50
					OR = {
						has_trait = ambitious
						has_trait = arrogant
						has_trait = brave
					}
				}
			}
			random = {
				chance = 5
				scope:rp_merit_liege = {
					rp_mark_campaign_incident_cooldown_effect = { DAYS = 180 }
					add_character_flag = {
						flag = rp_pursuit_chain_cooldown
						days = 365
					}
					trigger_event = rp_merit.2020
				}
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = { rp_can_fire_campaign_incident_trigger = yes }
				scope:rp_campaign_actor = {
					has_variable = rp_merit_score
					var:rp_merit_score >= 60
					OR = {
						has_trait = brave
						has_trait = deceitful
					}
				}
			}
			random = {
				chance = 3
				scope:rp_merit_liege = {
					rp_mark_campaign_incident_cooldown_effect = { DAYS = 180 }
					trigger_event = rp_merit.2030
				}
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = { rp_can_fire_campaign_incident_trigger = yes }
				scope:rp_campaign_actor = {
					has_variable = rp_merit_score
					var:rp_merit_score >= 80
					OR = {
						has_trait = arrogant
						has_trait = ambitious
						has_trait = stubborn
					}
				}
			}
			random = {
				chance = 4
				scope:rp_merit_liege = {
					rp_mark_campaign_incident_cooldown_effect = { DAYS = 180 }
					trigger_event = rp_merit.2040
				}
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = { rp_can_fire_campaign_incident_trigger = yes }
				scope:rp_campaign_actor = {
					has_variable = rp_merit_score
					var:rp_merit_score >= 45
					OR = {
						has_trait = just
						has_trait = diligent
						has_trait = stubborn
					}
				}
			}
			random = {
				chance = 2
				scope:rp_merit_liege = {
					rp_mark_campaign_incident_cooldown_effect = { DAYS = 180 }
					trigger_event = rp_merit.2080
				}
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = { rp_can_fire_campaign_incident_trigger = yes }
				scope:rp_campaign_actor = {
					has_variable = rp_merit_score
					var:rp_merit_score >= 40
					OR = {
						has_trait = patient
						has_trait = calm
						martial >= 14
						learning >= 10
					}
				}
			}
			random = {
				chance = 5
				scope:rp_merit_liege = {
					rp_mark_campaign_incident_cooldown_effect = { DAYS = 180 }
					trigger_event = rp_merit.2095
				}
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = { rp_can_fire_campaign_incident_trigger = yes }
				scope:rp_campaign_actor = {
					has_variable = rp_merit_score
					var:rp_merit_score >= 50
					OR = {
						has_trait = brave
						has_trait = stubborn
						has_trait = zealous
						has_character_flag = rp_publicly_shamed
					}
				}
			}
			random = {
				chance = 3
				scope:rp_merit_liege = {
					rp_mark_campaign_incident_cooldown_effect = { DAYS = 180 }
					trigger_event = rp_merit.2050
				}
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = { rp_can_fire_campaign_incident_trigger = yes }
				scope:rp_campaign_actor = {
					OR = {
						has_trait = brave
						has_trait = compassionate
						has_trait = honest
					}
					opinion = {
						target = scope:rp_merit_liege
						value >= 60
					}
				}
			}
			random = {
				chance = 8
				scope:rp_merit_liege = {
					rp_mark_campaign_incident_cooldown_effect = { DAYS = 180 }
					trigger_event = rp_merit.2060
				}
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = { rp_can_fire_minor_campaign_incident_trigger = yes }
			}
			random = {
				chance = 7
				scope:rp_merit_liege = {
					add_character_flag = {
						flag = rp_minor_campaign_incident_cooldown
						days = 45
					}
					trigger_event = rp_merit.2100
				}
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = { rp_can_fire_minor_campaign_incident_trigger = yes }
			}
			random = {
				chance = 6
				scope:rp_merit_liege = {
					add_character_flag = {
						flag = rp_minor_campaign_incident_cooldown
						days = 45
					}
					trigger_event = rp_merit.2101
				}
			}
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = { rp_can_fire_minor_campaign_incident_trigger = yes }
			}
			random = {
				chance = 6
				scope:rp_merit_liege = {
					add_character_flag = {
						flag = rp_minor_campaign_incident_cooldown
						days = 45
					}
					trigger_event = rp_merit.2102
				}
			}
		}
	}
}

rp_merit.2002 = {
	type = character_event
	hidden = yes

	immediate = {
		scope:war = { save_scope_as = rp_current_war }

		if = {
			limit = {
				scope:war = {
					primary_attacker = {
						rp_is_merit_system_enabled_trigger = yes
						OR = {
							this = root
							this = root.top_liege
						}
					}
				}
			}
			scope:war.primary_attacker = { save_scope_as = rp_merit_liege }
		}
		else_if = {
			limit = {
				scope:war = {
					primary_defender = {
						rp_is_merit_system_enabled_trigger = yes
						OR = {
							this = root
							this = root.top_liege
						}
					}
				}
			}
			scope:war.primary_defender = { save_scope_as = rp_merit_liege }
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				root = { rp_is_high_merit_campaign_actor_trigger = { LIEGE = scope:rp_merit_liege VALUE = 30 } }
			}
			root = { save_scope_as = rp_campaign_actor }
		}
		else_if = {
			limit = { exists = scope:rp_merit_liege }
			scope:rp_merit_liege = { rp_select_high_merit_campaign_actor_effect = { VALUE = 30 } }
		}

		if = {
			limit = {
				exists = scope:rp_merit_liege
				exists = scope:rp_campaign_actor
				scope:rp_merit_liege = {
					rp_can_fire_campaign_incident_trigger = yes
					NOT = { has_character_flag = rp_yang_zhen_chain_cooldown }
				}
				scope:rp_campaign_actor = {
					OR = {
						has_trait = vengeful
						has_trait = greedy
						has_trait = arrogant
					}
				}
			}
			random = {
				chance = 3
				scope:rp_merit_liege = {
					rp_mark_campaign_incident_cooldown_effect = { DAYS = 180 }
					add_character_flag = {
						flag = rp_yang_zhen_chain_cooldown
						days = 365
					}
					trigger_event = rp_merit.2010
				}
			}
		}
	}
}
