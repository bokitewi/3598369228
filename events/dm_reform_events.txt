dm_reform.0210 = {
	type = character_event
	title = dm_reform.0210.t
	desc = dm_reform.0210.desc
	theme = court
	left_portrait = root
	trigger = {
		exists = scope:dm_reform_story
		scope:dm_reform_story = { story_type = dm_reform_story }
		dm_reform_reformer_candidate_trigger = {
			RULER = scope:dm_reform_story.story_owner
		}
	}
	on_trigger_fail = {
		scope:dm_reform_story ?= { remove_variable = dm_reform_pending_reformer }
	}
	option = {
		name = dm_reform.0210.a
		scope:dm_reform_story = {
			dm_reform_install_reformer_effect = {
				REFORMER = root
			}
		}
		ai_chance = {
			base = 40
			opinion_modifier = {
				who = root
				opinion_target = scope:dm_reform_story.story_owner
				multiplier = 0.3
			}
			modifier = { add = 20 root = { has_trait = ambitious } }
			modifier = { add = 15 root = { has_trait = diligent } }
			modifier = { add = -25 root = { has_trait = content } }
			modifier = { add = -20 root = { has_trait = craven } }
		}
	}
	option = {
		name = dm_reform.0210.b
		scope:dm_reform_story = { remove_variable = dm_reform_pending_reformer }
		ai_chance = {
			base = 60
			modifier = { factor = 0 root = { has_trait = ambitious } }
		}
	}
}
