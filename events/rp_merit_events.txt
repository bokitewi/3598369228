namespace = rp_merit

rp_merit.0001 = {
	scope = combat_side
	hidden = yes

	immediate = {
		if = {
			limit = {
				combat.combat_war = {
					primary_attacker = { rp_is_merit_system_enabled_trigger = yes }
				}
				side_primary_participant = { is_attacker_in_war = root.combat.combat_war }
			}
			combat.combat_war.primary_attacker = { save_scope_as = rp_merit_liege }
		}
		else_if = {
			limit = {
				combat.combat_war = {
					primary_defender = { rp_is_merit_system_enabled_trigger = yes }
				}
				side_primary_participant = { is_defender_in_war = root.combat.combat_war }
			}
			combat.combat_war.primary_defender = { save_scope_as = rp_merit_liege }
		}

		side_commander ?= {
			save_scope_as = rp_primary_commander
			if = {
				limit = { rp_can_receive_merit_from_liege_trigger = { LIEGE = scope:rp_merit_liege } }
				if = {
					limit = { scope:wipe = yes }
					rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 24 }
					rp_check_major_battle_petition_effect = { LIEGE = scope:rp_merit_liege VALUE = 24 }
				}
				else = {
					rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 12 }
				}
				if = {
					limit = {
						OR = {
							martial >= 20
							has_trait = education_martial_4
						}
					}
					rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 3 }
				}
				rp_mark_war_service_participant_effect = yes
				rp_record_battle_stat_effect = yes
			}
		}

		every_side_commander = {
			limit = {
				rp_can_receive_merit_from_liege_trigger = { LIEGE = scope:rp_merit_liege }
				NOT = { this = scope:rp_primary_commander }
			}
			if = {
				limit = { scope:wipe = yes }
				rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 12 }
			}
			else = {
				rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 7 }
			}
			rp_mark_war_service_participant_effect = yes
			rp_record_battle_stat_effect = yes
		}

		every_side_knight = {
			limit = { rp_can_receive_merit_from_liege_trigger = { LIEGE = scope:rp_merit_liege } }
			if = {
				limit = { scope:wipe = yes }
				rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 6 }
			}
			else = {
				rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 3 }
			}
			rp_mark_war_service_participant_effect = yes
			rp_record_battle_stat_effect = yes
		}

		random_side_knight = {
			limit = { rp_can_receive_merit_from_liege_trigger = { LIEGE = scope:rp_merit_liege } }
			random = {
				chance = 35
				rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 8 }
				if = {
					limit = {
						OR = {
							prowess >= 18
							martial >= 16
							has_trait = brave
						}
					}
					rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 5 }
				}
				if = {
					limit = { scope:wipe = yes }
					rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 7 }
				}
				if = {
					limit = {
						root = {
							scope:rp_merit_liege = { is_attacker_in_war = root.combat.combat_war }
							combat.combat_war.primary_defender = { highest_held_title_tier >= tier_kingdom }
						}
					}
					rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 8 }
				}
				else_if = {
					limit = {
						root = {
							scope:rp_merit_liege = { is_defender_in_war = root.combat.combat_war }
							combat.combat_war.primary_attacker = { highest_held_title_tier >= tier_kingdom }
						}
					}
					rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 8 }
				}
				rp_record_capture_stat_effect = yes
				rp_check_major_battle_petition_effect = { LIEGE = scope:rp_merit_liege VALUE = 20 }
			}
		}
	}
}

rp_merit.1301 = {
	type = character_event
	title = rp_merit.1301.t
	theme = diplomacy
	override_background = { reference = throne_room_east }
	desc = rp_merit.1301.desc
	right_portrait = scope:rp_petition_liege

	trigger = {
		exists = scope:rp_petition_liege
		has_variable = rp_merit_score
		var:rp_merit_score >= 30
	}

	option = {
		name = rp_merit.1301.a
		trigger = {
			scope:rp_petition_liege = { gold >= 40 }
		}
		scope:rp_petition_liege = { remove_short_term_gold = 40 }
		add_gold = 40
		rp_consume_merit_effect = { VALUE = 10 }
		add_character_flag = {
			flag = rp_petition_recently_handled
			days = 180
		}
		scope:rp_petition_liege = {
			add_opinion = {
				target = root
				modifier = rp_merit_reward_gold_opinion
			}
		}
	}

	option = {
		name = rp_merit.1301.b
		if = {
			limit = { NOT = { has_trait = rp_merit_honor_1 } }
			add_trait = rp_merit_honor_1
		}
		add_character_modifier = {
			modifier = rp_merit_honored_modifier
			years = 10
		}
		rp_consume_merit_effect = { VALUE = 10 }
		add_character_flag = {
			flag = rp_petition_recently_handled
			days = 180
		}
		scope:rp_petition_liege = {
			add_opinion = {
				target = root
				modifier = rp_merit_reward_honor_opinion
			}
		}
	}

	option = {
		name = rp_merit.1301.c
		trigger = {
			var:rp_merit_score >= 60
			highest_held_title_tier >= tier_county
			highest_held_title_tier < tier_kingdom
			scope:rp_petition_liege = { rp_has_grantable_duchy_trigger = yes }
		}
		scope:rp_petition_liege = {
			random_held_title = {
				limit = {
					tier = tier_duchy
					is_landless_type_title = no
					NOT = { this = scope:rp_petition_liege.primary_title }
					NOT = { this = scope:rp_petition_liege.capital_county.de_jure_liege }
					any_de_jure_county = {
						holder = scope:rp_petition_liege
						NOT = { this = scope:rp_petition_liege.capital_county }
					}
				}
				create_title_and_vassal_change = {
					type = granted
					save_scope_as = change
					add_claim_on_loss = no
				}
				random_de_jure_county = {
					limit = {
						holder = scope:rp_petition_liege
						NOT = { this = scope:rp_petition_liege.capital_county }
					}
					change_title_holder = {
						holder = root
						change = scope:change
						take_baronies = no
					}
				}
				change_title_holder = {
					holder = root
					change = scope:change
				}
				resolve_title_and_vassal_change = scope:change
			}
			root = { rp_correct_reward_government_effect = yes }
		}
		set_variable = { name = rp_merit_score value = 0 }
		add_character_flag = {
			flag = rp_petition_recently_handled
			days = 365
		}
	}

	option = {
		name = rp_merit.1301.e
		trigger = {
			var:rp_merit_score >= 30
			can_add_hook = {
				target = scope:rp_petition_liege
				type = favor_hook
			}
		}
		add_hook = {
			target = scope:rp_petition_liege
			type = favor_hook
		}
		rp_consume_merit_effect = { VALUE = 8 }
		add_character_flag = {
			flag = rp_petition_recently_handled
			days = 180
		}
	}

	option = {
		name = rp_merit.1301.f
		trigger = {
			var:rp_merit_score >= 90
			can_add_hook = {
				target = scope:rp_petition_liege
				type = strong_obligation_hook
			}
		}
		add_hook = {
			target = scope:rp_petition_liege
			type = strong_obligation_hook
		}
		rp_consume_merit_effect = { VALUE = 25 }
		add_character_flag = {
			flag = rp_petition_recently_handled
			days = 180
		}
	}

	option = {
		name = rp_merit.1301.d
		add_character_modifier = {
			modifier = rp_merit_resentful_modifier
			years = 3
		}

rp_merit.0002 = {
	type = character_event
	hidden = yes

	immediate = {
		if = {
			limit = {
				scope:war = {
					primary_attacker = {
						rp_is_merit_system_enabled_trigger = yes
						OR = {
							this = root
							this = root.top_liege
						}
					}
				}
			}
			scope:war.primary_attacker = { save_scope_as = rp_merit_liege }
		}
		else_if = {
			limit = {
				scope:war = {
					primary_defender = {
						rp_is_merit_system_enabled_trigger = yes
						OR = {
							this = root
							this = root.top_liege
						}
					}
				}
			}
			scope:war.primary_defender = { save_scope_as = rp_merit_liege }
		}
		if = {
			limit = {
				exists = scope:war
				exists = scope:rp_merit_liege
				scope:rp_merit_liege = { rp_is_merit_system_enabled_trigger = yes }
			}
			if = {
				limit = { rp_can_receive_merit_from_liege_trigger = { LIEGE = scope:rp_merit_liege } }
				rp_add_merit_effect = { LIEGE = scope:rp_merit_liege VALUE = 10 }
				rp_record_siege_stat_effect = yes
				rp_mark_war_service_participant_effect = yes
			}
			else_if = {
				limit = { this = scope:rp_merit_liege }
				every_knight = {
					limit = {
						rp_can_receive_merit_from_liege_trigger = { LIEGE = root }
						has_character_flag = rp_war_service_participant
					}
					rp_add_merit_effect = { LIEGE = root VALUE = 2 }
					rp_record_siege_stat_effect = yes
				}
				every_vassal_or_below = {
					limit = {
						rp_can_receive_merit_from_liege_trigger = { LIEGE = root }
						has_character_flag = rp_war_service_participant
					}
					rp_add_merit_effect = { LIEGE = root VALUE = 2 }
					rp_record_siege_stat_effect = yes
				}
			}
		}

		if = {
			limit = {
				exists = scope:previous_controller
				scope:previous_controller = {
					OR = {
						rp_can_receive_merit_from_liege_trigger = { LIEGE = scope:war.primary_attacker }
						rp_can_receive_merit_from_liege_trigger = { LIEGE = scope:war.primary_defender }
					}
				}
			}
			if = {
				limit = {
					scope:war.primary_attacker = { rp_is_merit_system_enabled_trigger = yes }
					scope:previous_controller = { rp_can_receive_merit_from_liege_trigger = { LIEGE = scope:war.primary_attacker } }
				}
				scope:previous_controller = {
					rp_add_merit_effect = { LIEGE = scope:war.primary_attacker VALUE = -10 }
					rp_record_siege_stat_effect = yes
					rp_check_major_penalty_effect = { LIEGE = scope:war.primary_attacker VALUE = -10 }
				}
			}
			else_if = {
				limit = {
					scope:war.primary_defender = { rp_is_merit_system_enabled_trigger = yes }
					scope:previous_controller = { rp_can_receive_merit_from_liege_trigger = { LIEGE = scope:war.primary_defender } }
				}
				scope:previous_controller = {
					rp_add_merit_effect = { LIEGE = scope:war.primary_defender VALUE = -10 }
					rp_record_siege_stat_effect = yes
					rp_check_major_penalty_effect = { LIEGE = scope:war.primary_defender VALUE = -10 }
				}
			}
		}
	}
}
