# Brotherhood Betrayed Event - Fires for the betrayed character when relation is broken
sworn_brotherhood_events.0003 = {
    type = character_event
    title = sworn_brotherhood_events.0003.t
    desc = {
        desc = sworn_brotherhood_events.0003.desc.intro
        first_valid = {
            triggered_desc = {
                trigger = { has_trait = just }
                desc = sworn_brotherhood_events.0003.desc.just_betrayed
            }
            triggered_desc = {
                trigger = { has_trait = wrathful }
                desc = sworn_brotherhood_events.0003.desc.wrathful_betrayed
            }
            triggered_desc = {
                trigger = { has_trait = cynical }
                desc = sworn_brotherhood_events.0003.desc.cynical_betrayed
            }
            desc = sworn_brotherhood_events.0003.desc.generic_betrayed
        }
        desc = sworn_brotherhood_events.0003.desc.outro
    }
    theme = friendship_negative  # ← FIXED: Valid CK3 theme

    left_portrait = {
        character = root
        animation = anger
    }

    right_portrait = {
        character = scope:actor  # ← FIXED: Use scope:actor instead of FROM
        animation = shame
    }

    # This event fires automatically via on_relation_blood_brother_removed
    fire_only_once = yes  # ← FIXED: CK3 syntax

    immediate = {
        scope:actor = { save_scope_as = betrayer }  # ← FIXED: Save the other character
    }

    option = { # Righteous Indignation (Just/Honest)
        name = sworn_brotherhood_events.0003.a
        trigger = {
            OR = {
                has_trait = just
                has_trait = honest
            }
        }

        add_stress = 50
        scope:betrayer = {
            add_opinion = {
                target = root
                modifier = opinion_betrayed_me
                years = 10
            }
        }
        add_trait = cynical
        remove_trait = trusting

        scope:betrayer = {
            add_character_modifier = {
                modifier = oath_breaker_modifier
                years = 15
            }
        }
    }

    option = { # Vengeful Wrath (Wrathful/Brave)
        name = sworn_brotherhood_events.0003.b
        trigger = {
            OR = {
                has_trait = wrathful
                has_trait = brave
            }
        }

        add_rival = scope:betrayer
        add_claim_on_liege_title_of = scope:betrayer

        scope:betrayer = {
            add_character_modifier = {
                modifier = dishonorable_modifier
                years = 20
            }
        }
    }

    option = { # Cynical Acceptance (Cynical/Ambitious)
        name = sworn_brotherhood_events.0003.c
        trigger = {
            OR = {
                has_trait = cynical
                has_trait = ambitious
            }
        }

        add_stress = 15
        scope:betrayer = {
            add_opinion = {
                target = root
                modifier = opinion_expected_betrayal
                years = 5
            }
        }
        add_prestige = 50
    }
}

# Oath Broken Event - Fires for the betrayer character
sworn_brotherhood_events.0004 = {
    type = character_event
    title = sworn_brotherhood_events.0004.t
    desc = sworn_brotherhood_events.0004.desc
    theme = guilt

    left_portrait = {
        character = root
        animation = shame
    }

    trigger = {
        exists = scope:actor
        had_relation_blood_brother = { target = scope:actor }  # ← FIXED: Use scope:actor
    }

    option = { # Regret the Betrayal
        name = sworn_brotherhood_events.0004.a

        add_stress = 30
        add_piety = -100
        scope:actor = {
            add_opinion = {
                target = root
                modifier = opinion_self_guilt
                years = 5
            }
        }
    }

    option = { # Embrace the Ambition
        name = sworn_brotherhood_events.0004.b
        trigger = {
            OR = {
                has_trait = ambitious
                has_trait = cynical
            }
        }

        add_stress = 10
        add_prestige = 75
        scope:actor = {
            add_opinion = {
                target = root
                modifier = opinion_cold_ambition
                years = 10
            }
        }
    }
}
